\section{Módulo: \texttt{data/universe/build\_universe.py}}
\label{sec:build_universe}

\subsection{Objetivo y filosofía de diseño}
\label{sec:build_universe:objective}

El módulo \texttt{build\_universe.py} construye el \textbf{universo invertible point‑in‑time (PIT) diario de small caps de EE.\ UU.}, con el objetivo de:

\begin{itemize}
  \item \textbf{Evitar survivorship bias} en el diseño de la base de datos.
  \item Mantener \textbf{trazabilidad completa} de por qué un ticker entra o sale del universo.
  \item Soportar \textbf{reproducibilidad} y \textbf{auditabilidad} industrial, compatible con pipelines de investigación quant y backtesting industrial.
\end{itemize}

La filosofía de diseño sigue principios análogos a los utilizados en índices institucionales (\textit{e.g.} MSCI Global Investable Market \cite{msci2024}, Russell 2000 \cite{lseg2024}) y en frameworks de universo PIT utilizados en fondos sistemáticos \cite{quantjourney2025}.
\begin{itemize}
  \item \textbf{PIT}: los datos de \textit{market cap}, liquidez y precio usados en la fecha de decisión \(t\) no pueden incorporar información futura.
  \item \textbf{Inversión parametrizable}: el universo se construye a partir de un \texttt{config.yaml} versionado, con \texttt{config\_hash} asociado.
  \item \textbf{Explicabilidad}: toda exclusión debe estar acompañada de una \texttt{eligibility\_reason} codificada.
\end{itemize}

\subsection{Definición formal del universo invertible}
\label{sec:build_universe:formal}

Sea \(\mathcal{T}\) el conjunto de títulos listados en EE.\ UU., con \(\mathcal{E} \subseteq \mathcal{T}\) el subconjunto de \textit{common stocks} elegibles por tipo de instrumento. Sea \(t \in \mathcal{D}\) una fecha de sesión en el calendario de mercado, y \(\mathcal{U}_t \subseteq \mathcal{E}\) el universo invertible en esa fecha.

\paragraph{Entradas}
\label{par:build_universe:inputs}
\begin{itemize}
  \item \(\mathcal{D} = [t_0, t_1]\): rango de fechas de sesión \texttt{start\_date, end\_date}.
  \item \(\mathcal{L}\): maestro de listados, con atributos por ticker \(s \in \mathcal{T}\):
  \begin{itemize}
    \item \(\mathtt{symbol}\), \(\mathtt{listing\_exchange}\), \(\mathtt{security\_type}\),
    \item \(\mathtt{list\_date}\), \(\mathtt{delist\_date}\) (nullable),
    \item \(\mathtt{status}\) (active/inactive, paused/suspended, etc.).
  \end{itemize}
  \item \(\mathcal{M}_t\): \textit{market cap} ajustado en USD a la fecha \(t\).
  \item \(\mathcal{V}_t\): \textit{ADV20} en USD o acciones a la fecha \(t\).
  \item \(\mathcal{P}_t\): precio de cierre ref \(p_{t,s}\) en fecha \(t\).
  \item \(\mathcal{C}\): calendario de mercado (fechas de sesión).
\end{itemize}

\paragraph{Reglas de negocio (configurables)}
\label{par:build_universe:rules}
Sea \(r = \{\text{rules}\}\) un conjunto de reglas parametrizables (definido en \texttt{universe\_config.yaml}):

\begin{itemize}
  \item \textbf{Tipo de instrumento}:
  \[
    s \in \mathcal{E} \Leftrightarrow \texttt{security\_type}(s) = \textsc{common\_stock} \wedge \text{no ADR/ETF/pref/SPAC/warrant}^r
  \]
  \item \textbf{Bolsa/mercado}:
  \[
    s \in \mathcal{E} \Rightarrow \mathtt{listing\_exchange}(s) \in \mathcal{E}_{\text{US}}
  \]
  \item \textbf{Small cap}:
  \[
    s \in \mathcal{E}_{\text{small}} \Leftrightarrow \mathtt{market\_cap\_usd}(s,t) \in [\underline{M}, \overline{M}] \ \text{or} \ \text{percentil}^r
  \]
  \item \textbf{Precio mínimo}:
  \[
    p_{t,s} \geq p_{\min}
  \]
  \item \textbf{Liquidez mínima}:
  \[
    \mathtt{adv20\_usd}(s,t) \geq v_{\min}
  \]
  \item \textbf{Antigüedad mínima} (opcional):
  \[
    t - \mathtt{list\_date}(s) \geq \Delta t_{\min}
  \]
\end{itemize}

La definición de universo condicional a la información disponible hasta \(t\) (o \(t-1\), según ejecución):

\[
  s \in \mathcal{U}_t \ : \ 
  \begin{cases}
    s \in \mathcal{E} \\
    \mathtt{list\_date}(s) \leq t \leq \mathtt{delist\_date}(s) \vee \texttt{delist\_date}(s) = \mathtt{null} \\
    \mathtt{p\_min}^r(p_{t,s}) \\
    \mathtt{v\_min}^r(\mathtt{adv20}(s,t)) \\
    \mathtt{cap\_band}^r(\mathtt{mkt\_cap}(s,t)) \\
    \text{reglas especiales de mercado/estado}
  \end{cases}
\]

\subsection{Arquitectura de datos y anti‑bias PIT}
\label{sec:build_universe:pit}

\paragraph{Panel PIT \texttt{date x symbol}}
\label{par:build_universe:panel}
A partir de \(\mathcal{L}\) y \(\mathcal{C}\), se construye un panel \(\mathcal{P} = \{(t,s)\}\) tal que:

\[
  (t,s) \in \mathcal{P} \Leftrightarrow 
  \mathtt{list\_date}(s) \leq t \leq \max\{\mathtt{delist\_date}(s), t_{\max}\}
\]
donde \(t_{\max}\) es el máximo de \(\mathcal{D}\). Esto garantiza:
\begin{itemize}
  \item ningún ticker aparece antes de \(\mathtt{list\_date}(s)\),
  \item los tickers \textit{delisted} se mantienen hasta su fecha de deslistado.
\end{itemize}

\paragraph{Anti‑bias por construcción}
\label{par:build_universe:anti_bias}
El diseño impide los principales \textit{survivorship bias}:
\begin{itemize}
  \item \textbf{Exclusión temprana de delistados}: \(\mathtt{delist\_date}(s)\) se usa para borrar \((t,s)\) a partir de \(t > \mathtt{delist\_date}(s)\).
  \item \textbf{Exclusión de IPOs prematuras}: \(\Delta t_{\min}\) en listado.
  \item \textbf{Sin backfill futuro}: \(\mathcal{M}_t, \mathcal{V}_t, \mathcal{P}_t\) se obtienen únicamente con datos disponibles hasta \(t\) (o \(t-1\)).
\end{itemize}

\subsection{Enriquecimiento con features PIT}
\label{sec:build_universe:enrichment}

\paragraph{Market cap y liquidez}
\label{par:build_universe:capvol}
Dado \(\mathcal{P}\), se enriquece con:
\begin{itemize}
  \item \(\mathtt{market\_cap\_usd}(s,t)\),
  \item \(\mathtt{adv20\_usd}(s,t)\),
  \item \(\mathtt{close\_ref}(s,t)\).
\end{itemize}

Si no existe valor PIT para \(\mathtt{mkt\_cap}\) o \(\mathtt{adv20}\), se aplica una política configurable:
\[
  \mathtt{strategy\_missing}(r) :
  \begin{cases}
    \text{excluir}      & \text{(default)} \\
    \text{forward\_fill} & \text{(ventana temporal limitada)} \\
    \text{proxy}         & \text{(por sector/market cap)}
  \end{cases}
\]

\subsection{Aplicación de filtros en orden determinista}
\label{sec:build_universe:filter_order}

Los filtros aplicados forman una secuencia determinista \(\mathcal{F} = \{F_1, F_2, \dots, F_k\}\) para cada \((t,s)\):

\begin{enumerate}
  \item \textbf{Tipo de instrumento}: \(F_1(s)\).
  \item \textbf{Bolsa/mercado}: \(F_2(s)\).
  \item \textbf{Ventana de vida}: \(F_3(t,s)\).
  \item \textbf{Precio mínimo}: \(F_4(p_{t,s})\).
  \item \textbf{Liquidez mínima}: \(F_5(\mathtt{adv20}(s,t))\).
  \item \textbf{Banda de small cap}: \(F_6(\mathtt{mkt\_cap}(s,t))\).
  \item \textbf{Reglas especiales}: securities halted, flags regulatorios, etc.
\end{enumerate}

Para cada \((t,s)\):
\[
  \mathtt{eligibility}(t,s) := \prod_{i=1}^k \mathbb{I}(F_i(t,s) = \texttt{True})
\]
Si \(\mathtt{eligibility}(t,s) = 0\), se registra \(\mathtt{eligibility\_reason}(t,s)\) codificando el primer filtro que falla (o todos, según detalle deseado).

\subsection{Estado de membresía y tracking devida}
\label{sec:build_universe:states}

Se definen estados discretos \( \mathtt{state}(t,s) \in \Omega_{\text{state}} \):

\begin{itemize}
  \item \texttt{active}            : si \(\mathtt{eligibility}(t,s) = 1\).
  \item \texttt{halted}           : si \(s\) está suspendido en mercado.
  \item \texttt{delisted}         : si \(t > \mathtt{delist\_date}(s)\).
  \item \texttt{excluded\_by\_rule} : si \(F_i(t,s) = 0\) para algún \(i\).
\end{itemize}

La historia de \(\mathtt{state}(t,s)\) se persiste en \texttt{parquet} para auditoría de \textit{turnover} y \textit{survivorship}.

\subsection{Outputs y estructura de archivos}
\label{sec:build_universe:outputs}

\paragraph{Historia diaria}
\label{par:build_universe:hist}
\begin{itemize}
  \item Ruta:
  \[
    \texttt{data/universe/history/date=\{yyyy-mm-dd\}/part-*.parquet}
  \]
  \item Schema mínimo:
  \begin{itemize}
    \item \texttt{date}                  : fecha de sesión.
    \item \texttt{symbol}                : ticker.
    \item \texttt{is\_eligible}          : \(\in \{0,1\}\).
    \item \texttt{eligibility\_reason}   : \texttt{enum} o \texttt{string} codificada.
    \item \texttt{market\_cap\_usd}      : \texttt{float} (nullable).
    \item \texttt{adv20\_usd}            : \texttt{float} (nullable).
    \item \texttt{close\_ref}            : \texttt{float} (nullable).
    \item \texttt{listing\_exchange}     : \texttt{string}.
    \item \texttt{security\_type}        : \texttt{string}.
    \item \texttt{in\_universe\_rank}    : ranking por \texttt{market\_cap} o \texttt{liquidity} (opcional).
    \item \texttt{run\_id}, \texttt{config\_hash}, \texttt{ingest\_ts\_utc}.
  \end{itemize}
\end{itemize}

\paragraph{Snapshots y artefactos}
\label{par:build_universe:snapshots}
\begin{itemize}
  \item \texttt{data/universe/current.parquet}:
  \[
    \mathcal{U}_{t_{\max}} \cup \text{campos de tracking}
  \]
  \item \texttt{data/universe/constituents\_\{date\}.parquet}:
  \[
    \{(s,t) \in \mathcal{P} \mid \mathtt{eligibility}(t,s) = 1\}
  \]
  \item \texttt{data/universe/exclusions\_\{date\}.parquet}:
  \[
    \{(s,t) \in \mathcal{P} \mid \mathtt{eligibility}(t,s) = 0\}
  \]
  \item \texttt{universe\_manifest.json}:
  \[
    \{(t, |\mathcal{U}_t|), \text{conteo de exclusiones por motivo}, \text{checksums}\}
  \]
  \item \texttt{universe\_counts\_by\_day.parquet}:
  \[
    \{t, |\mathcal{U}_t|, |\mathcal{E}_t|\}
  \]
  \item \texttt{universe\_turnover\_stats.parquet}:
  \[
    \text{turnover}(t) := \frac{|\mathcal{U}_t \setminus \mathcal{U}_{t-1}| + |\mathcal{U}_{t-1} \setminus \mathcal{U}_t|}{|\mathcal{U}_{t-1} \cup \mathcal{U}_t|}
  \]
\end{itemize}

\subsection{Validación y observabilidad}
\label{sec:build_universe:validations}

\paragraph{Integridad de datos}
\label{par:build_universe:integrity}
\begin{itemize}
  \item \textbf{Sin duplicados}:
  \[
    \forall t_1 \neq t_2 \text{ o } s_1 \neq s_2 \Rightarrow (t_1,s_1) \neq (t_2,s_2)
  \]
  \item \textbf{Campo no nulo}:
  \[
    \mathtt{is\_eligible} \in \{0,1\}, \quad \text{siempre disponible}
  \]
  \item \textbf{Motivo de exclusión}:
  \[
    \mathtt{eligibility\_reason}(t,s) \text{ presente} \iff \mathtt{is\_eligible}(t,s) = 0
  \]
\end{itemize}

\paragraph{Consistencia temporal}
\label{par:build_universe:temporal_consistency}
\begin{itemize}
  \item \textbf{Antes de listado}:
  \[
    \forall (t,s) \in \mathcal{P}: t \geq \mathtt{list\_date}(s)
  \]
  \item \textbf{Después de deslistado}:
  \[
    t > \mathtt{delist\_date}(s) \Rightarrow s \notin \mathcal{U}_t
  \]
  \item \textbf{Jump de tamaño}:
  \[
    \Delta \mathcal{U}_t := |\mathcal{U}_t| - |\mathcal{U}_{t-1}|
  \]
  se reporta como \texttt{WARNING} si \(|\Delta \mathcal{U}_t| > \tau_{\text{jump}}\).
\end{itemize}

\paragraph{Sanidad de filtros}
\label{par:build_universe:sanity_checks}
\begin{itemize}
  \item \(\mathtt{market\_cap\_usd}(s,t) \geq 0\) si no nulo.
  \item \(\mathtt{adv20\_us
