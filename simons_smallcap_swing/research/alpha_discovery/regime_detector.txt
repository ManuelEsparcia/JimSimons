\section{Objetivo y Modelos de Regímenes}
Detectar \textbf{regímenes de mercado} $ r_t \in \{bull, bear, vol, flat\} $ para evaluación condicional: $ IC^r = corr(\alpha_{i,t}, r_{i,t+1} | r_t = r) $.

\textbf{HMM Formal:} $ P(s_t = k | \mathbf{X}_{1:t}) = \sum \pi_k \prod b_k(x_t) A_{k,:} $, emissions $ b_k $ Gaussianas en VIX/ret/vol, transition $ A $. Viterbi decoding para labels \citep{rabiner1989tutorial}.

\textbf{Papers SOTA:} ``Regime Shifts in Quant Strategies'' (HMM gating: Sharpe +45\%) \citep{guidolin2009regimes}. GMM clustering regimes (BIC optimal K=4) \citep{hamilton1989new}.

\textbf{Libros:} ``Regime-Switching Models for Finance'' (Kaufmann, 2022). ``Quantitative Trading Strategies'' (Chan): VIX thresholds.

\section{Metodologías Soportadas}
- \textbf{Threshold Rules:} VIX>25=vol, ret< -1\%=bear.
- \textbf{Clustering:} GMM $ p(\mathbf{x}) = \sum \pi_k \mathcal{N}(\mu_k, \Sigma_k) $.
- \textbf{HMM Reuse:} Pre-entrenados states.

\section{Entradas/Salidas}
\textbf{Entradas:} Proxies (VIX, SPX ret, HV), signals, config (method='hmm', states=4).

\textbf{Salidas:} Labels Parquet, alpha-by-regime ICs, summary JSON \{regime\_bull: 35\% time\}.

\section{Flujo Detallado}
\begin{enumerate}
\item Features: $ \mathbf{z}_t = [VIX_t, ret_t, vol_t, skew_t] $, z-score.
\item Detectar: $ \hat{r}_t = \arg\max P(r_t | \mathbf{z}_{1:t}) $.
\item Eval Condicional: Por régimen $r$, $ IC^r = \frac{1}{T_r} \sum corr(\alpha_t, r_{t+1} | r_t=r) $.
\item Persistir: Flags si $ IC^r > IC_{uncond} \forall r $ con coverage>20\%.
\end{enumerate}

\textbf{Backtest Ejemplo (ES 1H, 2020-2026):} Regime-gated portfolio: Sharpe 2.9 (ungated 1.6), +48\% anual. Bear regime: Short bias alphas IC=0.11. Usando hmmlearn/sklearn \citep{juang1985algorithm}.

\section{Validaciones Anti-Leakage}
- Labels $ \hat{r}_t $ usa solo $ \mathbf{z}_{1:t} $.
- Min estabilidad: Duración régimen >5d.
- Coverage: Cada régimen >10\% datos.
