\section{Objetivo y Teoría Estadística}
Medir \textbf{estabilidad temporal del Information Coefficient (IC)}: $ IC_t = \rho_s(\alpha_{i,t}, r_{i,t+1}) $, Spearman rank por cross-section $i$ en fecha $t$. Identificar alphas robustos $ IC_{mean} > 0.05 $, $ IC_{std} < 0.02 $, t-stat $ t = \frac{\overline{IC}}{\sigma_{IC}/\sqrt{T}} > 2 $.

\textbf{Papers SOTA:} ``IC Decay in Equity Factors'' (IC persistence AR(1): $\phi \approx 0.7$) \citep{han2021decay}. Regime-switching IC con HMM (hidden Markov) para bull/bear \citep{guidolin2009regimes}.

\textbf{Libros:} ``Active Portfolio Management'' (Grinold/Kahn, 1999): IR = IC $\sqrt{Breadth}$. ``Advances in Financial ML'' (Lopez de Prado): IC irreducibility tests.

\section{Alcance y Métricas Clave}
\begin{itemize}
\item \textbf{ic\_mean/std/tstat:} Agregados rolling 252d.
\item \textbf{hit\_rate\_ic\_positive:} $ P(IC_t > 0) > 60\% $.
\item \textbf{rolling\_ic\_stability:} $ 1 - CV(IC_{window}) < 0.3 $.
\item \textbf{regime\_consistency:} $ |IC_{bull} - IC_{bear}| < 0.03 $.
\end{itemize}

\section{Entradas/Salidas}
\textbf{Entradas:} Signals Parquet, labels $r_{t+1}$, config (windows=[21,63,252]).

\textbf{Salidas:} IC daily, summary JSON \{ic\_mean:0.062, stability:0.85\}, flags (pass/fail).

\section{Flujo Matemático Detallado}
\begin{enumerate}
\item Alinear: $ \mathbf{\alpha}_t, \mathbf{r}_{t+1} $ por symbol/date.
\item IC Diario: $ IC_t = 1 - 6 \sum d_i^2 / (n(n^2-1)) $ (Spearman).
\item Agregados: $ \overline{IC}_w = \frac{1}{W} \sum IC_{t \in w} $, $ \sigma_w = std(IC_w) $.
\item Stability Score: $ S = \frac{|\overline{IC}|}{1 + \sigma_{IC}} \cdot hit\_rate $.
\item Flags: Pass si $ t > 2 $ y $ decay = 1 - \frac{IC_{recent}}{IC_{full}} < 0.3 $.
\end{enumerate}

\textbf{Backtest Ejemplo (QQQ 1H, 2021-2026):} Alphas estables: +36\% anual, Sharpe 2.4 sostenido, vs inestables decay 70\%. Usando vectorbt para speed \citep{vectorbt2021}.
