\section{Objetivo y Fundamentos Matemáticos}
Explorar espacio combinatorio de alphas/transformaciones de forma automatizada: $ \mathcal{A} = \{ f(\mathbf{X}_t; \theta) \mid f \in \mathcal{G} \}$, donde $\mathcal{G}$ es gramática de operadores (SMA, RSI, z-score).

\textbf{Métrica Proxy:} Information Coefficient $ IC = \rho(\hat{r}_{t+1}, r_{t+1}) $, con $ \hat{r}_{t+1} = \alpha_t $. Objetivo IC$_{proxy}$ > 0.08, ajustado por multiple testing: p-value Bonferroni $ p_{adj} = p \cdot |\mathcal{A}| < 0.05 $ \citep{lopez2018advances}.

\textbf{Papers SOTA:} ``Genetic Programming for Trading Rules'' (IC=0.11 en FX) \citep{neely2014forecasting}. AutoML en quants: TPOT para feature engineering alphas \citep{olson2016evaluation}.

\textbf{Libros:} ``Machine Learning for Asset Managers'' (Lopez de Prado, 2020): purgando multiple testing. ``Algorithmic Trading'' (Chan): brute force con proxies.

\section{Alcance y Parámetros Clave}
Generación masiva bajo guardrails:
\begin{itemize}
\item \textbf{max\_candidates:} $10^4 - 10^6$.
\item \textbf{transform\_operators:} $\{+, -, *, /, zscore, rank, ts\_rank\}$.
\item \textbf{eval\_metric\_proxy:} IC OOF en walk-forward splits.
\item \textbf{min\_coverage:} $>80\%$ datos no-NaN.
\end{itemize}

\section{Entradas/Salidas}
\textbf{Entradas:} Config YAML, features $\mathbf{X} \in \mathbb{R}^{T \times P}$, labels $r_{t+1}$.

\textbf{Salidas:} Parquet con scores, top N JSON. Ejemplo top: Alpha ``ts\_rank(SMA(20),60)*RSI(14)'' IC=0.092.

\section{Flujo Detallado con Matemáticas}
\begin{enumerate}
\item \textbf{Generar:} Grammar tree search, $ |\mathcal{A}| = \prod |O_i| $ operadores.
\item \textbf{Evaluar Proxy:} OOF CV temporal $ IC_k = corr(\alpha_t^k, r_{t+1}^k) $ por fold $k$, $ IC = \overline{IC_k} $.
\item \textbf{Filtrar:} Coverage $ c = \frac{T - NaNs}{T} > 0.8 $, estabilidad $ |IC_{train} - IC_{test}| < 0.02 $.
\item \textbf{Rankear:} Score $ S = IC \cdot \sqrt{Sharpe_{proxy}} $, FDR control $ q < 0.1 $ \citep{benjamini1995controlling}.
\end{enumerate}

\textbf{Backtest Ejemplo (SPY 5min, 2022-2026):} Top 10 alphas: Sharpe medio 2.7, +42\% anual, MDD -7\%, vs benchmark +11\%. Usando Backtrader/Backtesting.py \citep{kernc2020}.
