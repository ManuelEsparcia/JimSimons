\section{Objetivo y Fundamento Te\'orico}
Cuantificar contribuci\'on marginal de componentes arquitect\'onicos (calibration, ensemble, regime gating) identificando complejidad util vs innecesaria. En swing trading, priorizamos \textbf{Pareto-optimal}: m\'ax Sharpe con m\'inima latencia/opcost \cite{hull2018options}.

\textbf{Matem\'atica:} Beneficio neto de componente $C_k$:
\[
\Delta_k = \text{IR}(f_{\text{full}}) - \text{IR}(f_{-k}) - \lambda \cdot \Delta \text{Comp}(C_k)
\]
donde IR = Information Ratio, $\text{Comp} = \text{VC-dim} + \text{latencia (ms)} + \text{params (M)}$. Usamos Multi-Objective Optimization:
\[
\max_{\mathbf{a}} \, \left( \text{IR}(\mathbf{a}), \, -\text{Comp}(\mathbf{a}) \right), \quad \mathbf{a} \in \{0,1\}^K
\]
\cite{hooker2021rf}: Ensembles solo si $\Delta \text{RF} > 0.1$.

\section{Alcance y Limitaciones}
\textbf{Incluye:}
\begin{itemize}
    \item Ablation one-at-a-time: Calibration (Platt/ISOTonic), Ensemble (stacking/blender).
    \item Regime Gating: HMM/Threshold para regimes (bull/bear/vol-spike).
    \item Tradeoff: Performance vs Complexity (Pareto frontier).
    \item Arquitectura m\'inima robusta (Occam's razor quant).
\end{itemize}

\textbf{Excluye:} Feature ablation; manual redesign.

\section{Metodolog\'ia Estado del Arte}
\subsection{Protocolo}
1. Baseline: Full pipeline IR$_b$.
2. Ablated: $f_{-k}$, retrain en same splits (purged).
3. Pareto Score:
\[
\text{PS}_k = \frac{\Delta\text{IR}_k}{\Delta\text{Comp}_k + \epsilon}, \quad \text{KEEP if PS} > 1.
\]
4. Bootstrap CI para estabilidad OOS.

\textbf{VC-Dimension proxy:} Para LGBM ensemble: $d_{\text{VC}} \approx 10 \log(n_{\text{trees}} \cdot d_{\text{depth}})$.

\subsection{Research Inspiraci\'on}
\begin{itemize}
    \item \textbf{What If Ensembles Fail?} \cite{hooker2021rf}: Ablation RF para finance.
    \item \textbf{Complexity in Quant Models} \cite{ernest2020}: Chan extension.
    \item Regime Switching: Hamilton HMM \cite{hamilton1994time}.
    \item Industrial: Carver ``Systematic Trading'' para gating.
\end{itemize}

Papers swing: LSTM gating \cite{lstm_swing2022}; libros Hull \cite{hull2018options}, Chan \cite{chan2018quantitative}.

\section{Flujo y Anti-Leakage}
\begin{enumerate}
    \item \texttt{load\_baseline\_pipeline\_context()}.
    \item \texttt{build\_component\_plan()}: Variants grid.
    \item \texttt{run\_variant()}: Isolated change.
    \item \texttt{compute\_deltas()}: $\Delta$IR, $\Delta$DD, $\Delta$Comp.
    \item \texttt{score\_pareto()}: Frontier plot.
    \item \texttt{recommendations()}: Rule-based (PS >1.5 KEEP).
\end{enumerate}

\textbf{Anti-leakage:} Identical splits/objective; fixed risk/cost salvo ablated.

\section{Casos Especiales}
\begin{itemize}
    \item Acoplados: Joint ablation (e.g. gating+ensemble).
    \item Inestable: Flag FAIL, analyze variance.
    \item Marginal: Drop si $\Delta\text{IR} < 0.05$ y $\Delta\text{Comp} > 10\%$.
\end{itemize}

\textbf{Optimizaci\'on:} Meta-learning para auto-ablation; target: 30\% comp reduction sin IR loss.
