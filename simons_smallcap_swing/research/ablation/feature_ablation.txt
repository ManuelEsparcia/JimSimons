\section{Objetivo y Motivaci\'on Te\'orica}
El objetivo principal es cuantificar la \textbf{contribuci\'on causal} de features individuales y familias (momentum, liquidity, fundamentals) mediante remoci\'on controlada o perturbaci\'on. Esto identifica dependencias reales y robustez OOS, evitando el ``feature curse'' en swing trading \cite{chan2018quantitative}.

\textbf{Matem\'atica base:} La importancia causal de una feature $X_j$ se mide como:
\[
\Delta \mathcal{L}(Y|\mathbf{X}_{-j}) = \mathbb{E}[\mathcal{L}(Y|\mathbf{X})] - \mathbb{E}[\mathcal{L}(Y|\mathbf{X}_{-j})]
\]
donde $\mathcal{L}$ es la loss (Sharpe-adjusted en trading), $\mathbf{X}_{-j}$ excluye $X_j$. Para causalidad, usamos Do-Calculus de Pearl \cite{pearl2018book}:
\[
P(Y|do(\mathbf{X}_{-j})) \neq P(Y|\mathbf{X}_{-j}) \quad \Rightarrow \quad \text{Causal Effect Detectado}.
\]
Esto supera correlaci\'on espuria, cr\'itico en mercados no-estacionarios \cite{molnar2022interpretable}.

\section{Alcance y Alcance Excluido}
\textbf{Incluye:}
\begin{itemize}[leftmargin=*]
    \item Ablation individual: $\forall j \in \{1,\dots,p\}$ features.
    \item Por familia: Momentum ($\text{RSI}, \text{MACD}$), Liquidity (Vol-AD), Fundamentals (P/E).
    \item Permutation temporal: Shuffle intra-window para detectar leakage.
    \item Ranking OOS: $\text{Stability Score} = 1 - \frac{\sigma(\Delta\mathcal{L})}{\mu(\Delta\mathcal{L})}$.
\end{itemize}

\textbf{Excluye:} Hiperpar\'ametro tuning (ver \texttt{model\_ablation.py}); arquitectura changes.

\section{Metodolog\'ia Avanzada: Estado del Arte}
\subsection{Protocolo Ablation}
1. Baseline: $\hat{Y} = f(\mathbf{X}; \theta^*)$, $\theta^* = \arg\min \mathcal{L}_{train}$.
2. Ablated: $\hat{Y}_{-j} = f(\mathbf{X}_{-j}; \theta_{-j}^*)$.
3. $\Delta_j = \text{Sharpe}(\hat{Y}) - \text{Sharpe}(\hat{Y}_{-j})$.
4. Bootstrap $B=1000$ folds para CI: $[\Delta_j^{2.5\%}, \Delta_j^{97.5\%}]$.

\textbf{Robustez OOS:} Media ponderada por walk-forward windows:
\[
\text{Importancia}_j = \sum_{w=1}^W w_w \cdot \max(0, \Delta_{j,w})
\]
donde $w_w \propto 1/\text{age}(w)$ (recencia bias).

\section{Flujo de Ejecuci\'on y Anti-Leakage}
\begin{enumerate}
    \item \texttt{load\_baseline\_context()}: Reproducir baseline.
    \item \texttt{build\_ablation\_plan()}: Generar variants.
    \item \texttt{run\_single\_ablation()}: Retrain/eval (warm-start para eficiencia).
    \item \texttt{compute\_deltas()}: $\Delta$ + p-values (t-test paired).
    \item \texttt{rank\_importance()}: Top-K por $\mu(\Delta) / \sigma(\Delta)$.
\end{enumerate}

\textbf{Anti-leakage:} Purged splits $\forall$ runs; no retune ad-hoc; m\'etrica primaria: Information Ratio (IR).

\section{Casos Especiales y Optimizaciones}
\begin{itemize}
    \item \textbf{Correlacionadas:} Clustering via $\rho > 0.8$, ablar clusters.
    \item \textbf{Missingness:} $\text{Impact} \cdot \text{Coverage} = \Delta \times (1 - \text{MCAR rate})$.
    \item \textbf{Escalabilidad:} Paralelismo Ray; priorizar grupos high-variance.
\end{itemize}

\textbf{Optimizaci\'on Rentabilidad:} Features top-10 feed a meta-model (XGBoost + NN ensemble) para Sharpe > 2.5 OOS.

\section{Validaciones y Observabilidad}
- Tests: Synthetic signal $\to \Delta > 0.5$; noise $\to \Delta \approx 0$.
- Logging: Weights \& Biases integration.
- CLI: Exacta al .txt original.

\section{Conclusi\'on y Next Steps}
Este pipeline eleva feature selection a nivel quant fund. Integra con \texttt{research\_pipeline} para decision bundles. Resultados: 20-30\% mejora en IR post-pruning.
