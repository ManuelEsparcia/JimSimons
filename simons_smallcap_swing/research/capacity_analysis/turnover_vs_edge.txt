\section{Fundamentos Matemáticos}
Turnover: 
\begin{equation}
T_t = \frac{1}{2} \sum_i |w_{t,i} - w_{t-1,i}|
\end{equation}
Modelo Garleanu-Pedersen (2013):
\begin{equation}
T^* = \frac{A}{2(\lambda + \kappa)}, \quad \lambda = \frac{1}{\tau} \quad (\tau: holding\ period)
\end{equation}
Net edge: 
\begin{equation}
e_n(T) = e_g - \left( \gamma T + \frac{1}{2} \eta \sqrt{T \cdot ADV} \right)
\end{equation}
Ref: \cite{garleanu2013}; $\eta=0.1-1$ bps.

Sharpe óptimo: $\arg\max_T \frac{e_n(T)}{\sigma(T)}$.

\section{Flujo del Código (Pseudocódigo Optimizado con SciPy)}
\begin{lstlisting}
import pandas as pd
import numpy as np
from scipy.optimize import curve_fit
from scipy.stats import linregress

def cost_model(turnover, gamma=5e-4, eta=0.5e-3):  # bps
    return gamma * turnover + 0.5 * eta * np.sqrt(turnover)

# Carga
baseline = pd.read_parquet(baseline_backtest_path)
turnovers = np.array([0.2, 0.5, 1.0, 1.5, 2.0])  # Anual

results = []
for t in turnovers:
    # Simula throttled positions
    signals_throttled = baseline['signal'] * (1 - t / baseline['signal_strength'])
    gross_edge = baseline['gross_edge']
    costs = cost_model(t)
    net_edge = gross_edge - costs
    sharpe = net_edge / baseline['vol']
    results.append({'turnover': t, 'net_edge': net_edge.mean(), 'sharpe_net': sharpe.mean()})

df = pd.DataFrame(results)
df.to_parquet('curve.parquet')

# Fit paramétrico
def edge_curve(t, eg, g, e): return eg - (g*t + e*np.sqrt(t))
popt, _ = curve_fit(edge_curve, df['turnover'], df['net_edge'])
optimal_t = popt[0] / (2*popt[1])  # Aprox T*
\end{lstlisting}

\section{Estado del Arte Industrial y Mejoras}
- \textbf{Adaptive Turnover}: RL policy $\pi(T|s_t)$ con PPO (Proximal Policy Optimization), state=[vol, crowding, edge].
- \textbf{Hawkes Process}: Modela excitation $dN_t = \mu dt + \int e^{-\beta(t-s)} dN_s$ para cascades.
- \textbf{Joint Capacity}: $T^*_{joint} = \min(T^*, capacity / AUM)$.
Ref: López de Prado ``FML'' \cite{lopez2018}; Ang ``Asset Mgmt'' 2014 \cite{ang2014}.
