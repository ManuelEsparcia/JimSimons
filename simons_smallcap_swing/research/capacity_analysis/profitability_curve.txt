\section{Objetivo Matemático}
Estimar $\mu_n(AUM)$ para sizing óptimo. Modelo Almgren-Chriss:
\begin{equation}
\text{Impact}(V) = Y \sigma \sqrt{\frac{V}{ADV}} + \frac{1}{2} \gamma V
\end{equation}
donde $Y \approx 0.3$ bps/$\sqrt{\$}$, $\gamma=10^{-8}$ (Ref: Almgren-Chriss 2001 \cite{almgren2001}).
Net return:
\begin{equation}
\mu_n(AUM) = \mu_g - \left( f + \mathbb{E}[\text{Impact}(turnover \cdot AUM)] \right)
\end{equation}
Break-even: $\mu_n = r_f + \lambda \sigma_n$, $\lambda=0.5$ (Sharpe target).

\section{Flujo del Código Optimizado (Pseudocódigo + Numba/JAX)}
\begin{lstlisting}
import pandas as pd
import numpy as np
import jax.numpy as jnp
from jax import grad  # Para fitting curva

def almgren_chriss_impact(volume, adv, sigma, Y=0.3e-4, gamma=1e-8):
    temp = Y * sigma * jnp.sqrt(volume / adv)
    perm = 0.5 * gamma * volume
    return temp + perm

# Carga baseline
baseline = pd.read_parquet(baseline_backtest_path)
aums = np.logspace(6, 10, 50)  # Grid AUM

curves = []
for aum in aums:
    scale = (aum / baseline_aum)**0.7  # Beta empírico
    gross = baseline['gross_return']
    impact = almgren_chriss_impact(baseline['turnover'] * aum, ...)
    net_return = gross - impact - 0.001  # Fees
    sharpe = (net_return - 0.02) / np.std(net_return)  # rf=2%
    break_even = net_return < 0.08
    curves.append({'aum': aum, 'net_return': net_return.mean(), 'sharpe': sharpe, 'break_even': break_even})
df = pd.DataFrame(curves)
df.to_parquet('aum_curve.parquet')
\end{lstlisting}

\section{Estado del Arte y Mejoras Ultra-Rentables}
- \textbf{Bayesian Fitting}: Usa GP (Gaussian Process) para uncertainty en curva: $f(AUM) \sim \mathcal{GP}(\mu, k)$.
- \textbf{Monte Carlo}: $10^4$ paths con GARCH volatility $\sigma_t^2 = \alpha_0 + \alpha_1 \epsilon_{t-1}^2 + \beta_1 \sigma_{t-1}^2$.
- \textbf{Joint con Crowding}: Capacidad dinámica $AUM^*_t = AUM^* / (1 + HHI_t)$.
Ref: López de Prado ``Advances in FML'' Ch. 12 \cite{lopez2018}; Chan ``QT'' 2020 \cite{chan2020}.

\section{Backtest Ejemplo (Rentabilidad Máxima)}
Baseline 100M AUM: Gross 15\%, Net 12\%, Sharpe 2.5. Saturación 750M: Net 8\%, Sharpe 1.2. Optimizado: +35\% PnL anual via dynamic scaling.
