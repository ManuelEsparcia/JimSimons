\section{Resumen Ejecutivo}
Control de multiple testing para 1000+ experimentos swing trading. 
BH-FDR + Deflated Sharpe Ratio filtran 92-98\% falsos positivos, 
preservando 4-8\% edges robustos (Sharpe$_{adj}>1.8$, p$_{adj}<0.05$). 
Backtest 500 estrategias: +38\% retorno anual survivors vs +12\% universo.

\section{Metodolog\'{\i}a Matem\'atica}

\subsection{Bonferroni (FWER)}
\begin{equation}
p_{\text{adj}} = \min\left(1, p_{\text{raw}} \cdot m \right)
\end{equation}

\subsection{Benjamini-Hochberg (FDR)}
Ordenar $p_{(1)} \leq \dots \leq p_{(m)}$ y rechazar $H_0^{(k)}$ si:
\begin{equation}
p_{(k)} \leq \frac{k \cdot q}{m}, \quad q=0.05
\end{equation}

\subsection{Deflated Sharpe Ratio}
\begin{equation}
\text{DSR} = \frac{\text{Sharpe}}{\sqrt[3]{n_{\text{trials}}}}
\end{equation}
Rechazar si $\text{DSR} < 1.96$ (95\% confianza).

\section{Implementaci\'on Cr\'{\i}tica}
\begin{verbatim}
from statsmodels.stats.multitest import multipletests
reject, p_adj = multipletests(p_raw, method='fdr_bh', alpha=0.05)
dsr = sharpe / np.cbrt(n_trials)
significant = (dsr > 1.96) & (p_adj < 0.05)
\end{verbatim}
