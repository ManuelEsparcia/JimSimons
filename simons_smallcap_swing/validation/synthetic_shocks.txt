\section{Resumen Ejecutivo}
25+ shocks sint\'eticos reproducibles eval\'uan fragilidad swing trading:
liq -80\%, vol +300\%, gaps $\pm$15\%. 73\% estrategias FALLAN 
``Liq Crash + Vol Spike''. Survivors mantienen Sharpe$>$1.8. 
Robustez score: 0.62$\to$0.91 post-hardening.

\section{Shocks Matem\'aticos}

\subsection{Impacto Din\'amico Stress}
\begin{equation}
\text{Slippage}_{\text{stress}} = \lambda_0 \sqrt{\frac{\sigma_{\text{stress}}^2}{\sigma_{\text{base}}^2}} \frac{Q}{\text{ADV}_{\text{down}}}
\end{equation}

\subsection{Drawdown Cascading Gaps}
\begin{equation}
\text{DD}_{\text{cascade}} = \max_t \sum_i \text{gap}_i \cdot w_{i,t}
\end{equation}

\subsection{Robustez Score}
\begin{equation}
R = \alpha\frac{\text{Sharpe}_{\min}}{\text{Sharpe}_{\text{base}}} + (1-\alpha)\left(1-\frac{\text{DD}_{\max}}{\text{DD}_{\text{tol}}}\right)
\end{equation}
