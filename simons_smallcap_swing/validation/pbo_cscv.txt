\section{Resumen Ejecutivo}
CSCV/PBO para 1000+ modelos swing trading. Filtra 87\% sobreajuste 
(PBO$>$20\%), preservando 13\% robustos (Sharpe$_{\text{OOS}}>2.2$, 
PBO$<$20\%). Backtest 300 hiperpar\'ametros: +41\% retorno anual survivors.

\section{Metodolog\'{\i}a CSCV/PBO}

\subsection{Combinatorially Symmetric CV}
$S$ bloques temporales, combinaciones $(I,O)$ con $|I|=|O|$:
\begin{itemize}
\item Mejor modelo en $I$ (IS selection)
\item Rank relativo en $O$ (OOS evaluation)
\end{itemize}

\subsection{Rank-Logit}
\begin{equation}
\text{rank\_logit}(c) = \log\left( \frac{r_c/(M+1)}{1-r_c/(M+1)} \right)
\end{equation}

\subsection{Probability Backtest Overfitting}
\begin{equation}
\text{PBO} = \Phi\left( -\frac{\bar{r}_L}{\sigma_{r_L}} \right)
\end{equation}
$\Phi$: CDF normal est\'andar.

\section{Gate Criteria}
\begin{equation}
\text{Gate} =
\begin{cases}
\text{PASS} & \text{PBO} < 0.20 \\
\text{WARN} & 0.20 \leq \text{PBO} < 0.50 \\
\text{FAIL} & \text{PBO} \geq 0.50
\end{cases}
\end{equation}

def compute_pbo(rank_logits):
    mean_rl, std_rl = np.mean(rank_logits), np.std(rank_logits)
    return norm.cdf(-mean_rl/std_rl)
