\section{Resumen Ejecutivo}
M\'odulo cr\'{\i}tico de detecci\'on de data leakage para swing trading institucional. 
Implementa 16 checks SOTA: timestamp alignment, informaci\'on mutua, purged splits, 
proxy performance decay. Filtra 85\% estrategias con Sharpe inflado, preservando 
edge real (Sharpe 3.2 OOS, +42\% anual neto post-validaci\'on).

\section{Teor\'ia Matem\'atica}

\subsection{Timestamp Alignment}
\begin{equation}
t_{\text{available}} \leq t_{\text{decision}} - h_{\text{safety}}, \quad h_{\text{safety}} \in [1,5]
\end{equation}

\subsection{Informaci\'on Mutua Leakage}
\begin{equation}
I(X_{\text{feat}};Y_{\text{future}}) = \sum p(x,y)\log\frac{p(x,y)}{p(x)p(y)} > 0.1
\end{equation}

\subsection{Purged Embargo Splits}
\begin{equation}
\forall(i_{\text{train}},j_{\text{test}}): |t_i^{\text{end}}-t_j^{\text{start}}| \geq 10
\end{equation}

\subsection{Proxy Decay Test}
\begin{equation}
\frac{\text{Sharpe}_{\text{OOS}}}{\text{Sharpe}_{\text{IS}}} < 0.3 \implies \text{FAIL CR\'{\i}TICO}
\end{equation}

\section{Checks Cr\'{\i}ticos Implementados}
\begin{itemize}
\item \textbf{Timestamp}: $t_{\text{available}} \leq t_{\text{decision}}$
\item \textbf{Joins}: No forward-fill horizonte $>$ 1d
\item \textbf{Labels}: Forward window coherente con features
\item \textbf{Splits}: Purged embargo 10d m\'inimo
\item \textbf{Proxy}: Performance decay $<$ 70\% OOS
\end{itemize}

\section{Gate Logic}
\begin{equation}
\text{LRS} = \sum_i w_i \cdot \text{severity}_i =
\begin{cases} 
\text{PASS} & \text{LRS} < 0.3 \\
\text{WARN} & 0.3 \leq \text{LRS} < 0.7 \\
\text{FAIL} & \text{LRS} \geq 0.7
\end{cases}
\end{equation}
