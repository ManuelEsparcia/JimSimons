\section{Matem\'atica Walk-Forward Bias}

\subsection{Overlap Contamination}
\begin{equation}
\text{Overlap} = \frac{|W_{\text{test}_i} \cap W_{\text{test}_j}|}{\min(|W_i|,|W_j|)} < 0.05
\end{equation}

\subsection{Boundary Leakage}
\begin{equation}
\text{Boundary Bias} = \frac{1}{B} \sum_b (\text{Perf}_{\text{boundary}} - \text{Perf}_{\text{interior}})
\end{equation}

\subsection{Train-Test Gap}
\begin{equation}
\text{Gap Days}_i = t_{\text{train end}} - t_{\text{test start}} \geq 5-10
\end{equation}

\section{Implementaci\'on Cr\'{\i}tica}
\begin{verbatim}
def overlap_ratio(w1, w2):
    return len(set(w1)&set(w2))/min(len(w1),len(w2))

def boundary_bias(windows, perfs):
    boundary = [perfs[0], perfs[-1]]
    return np.mean(np.array(boundary)-np.mean(perfs[1:-1]))

flags = {'overlap': overlap_ratio>0.05, 
         'boundary': abs(boundary_bias)>0.2}
\end{verbatim}

\section{WF Bias Score}
\begin{equation}
\text{WFBS} = 0.4\cdot\text{Overlap} + 0.3\cdot|\text{Boundary}| + 
              0.2\cdot\text{Gap Rate} + 0.1\cdot\text{Instability}
\end{equation}
\begin{equation}
\text{Gate WFBS} =
\begin{cases}
\text{PASS} & <0.20 \\
\text{WARN} & [0.20, 0.50) \\
\text{FAIL} & \geq 0.50
\end{cases}
\end{equation}
