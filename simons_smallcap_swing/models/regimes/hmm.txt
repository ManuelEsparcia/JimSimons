\section{Hidden Markov Model (HMM) para detección de regímenes en swing trading}

\subsection{Objetivo del módulo}
El módulo \texttt{models/regimes/hmm.py} estima \textbf{estados latentes de mercado} mediante \textbf{Hidden Markov Models (HMM) gaussianos} sobre el vector de \texttt{regime_features}, con el objetivo de segmentar regímenes de mercado (p. ej. \emph{risk‑on}, \emph{risk‑off}, \emph{transitional}, \emph{crisis}) y habilitar una toma de decisiones adaptativa en un sistema de swing trading cuantitativo.

Formalmente, sea \(\mathbf{X}_t \in \mathbb{R}^d\) el vector de \texttt{regime_features} observables en el día \(t\) y \(S_t \in \{0,1,\dots,K-1\}\) una variable latente discreta que representa el \textbf{estado inobservable} de mercado en el instante \(t\). El modelo HMM gaussiano asume:
\begin{equation}
\mathbb{P}(S_0 = k) = \pi_k,
\end{equation}
\begin{equation}
\mathbb{P}(S_t = j \mid S_{t-1} = i) = A_{ij},
\end{equation}
\begin{equation}
\mathbf{X}_t \mid S_t = k \sim \mathcal{N}(\boldsymbol{\mu}_k, \boldsymbol{\Sigma}_k),
\end{equation}
donde \(k, j \in \{0,\dots,K-1\}\), \(\boldsymbol{\pi} \in \mathbb{R}^K\) es el vector de probabilidades iniciales, \(\mathbf{A} \in \mathbb{R}^{K \times K}\) la matriz de transiciones, y \(\{\boldsymbol{\mu}_k, \boldsymbol{\Sigma}_k\}_{k=0}^{K-1}\) parámetros de emisión gaussiana. Este modelo permite:
\begin{itemize}
  \item Segmentar dinámicamente el espacio de mercado en un número reducido de estados con características macroeconómicas bien definidas \cite{Hamilton1989, Ang2012}.
  \item Soportar políticas de \emph{regime‑gating} en estrategias de swing trading, ajustando exposición y leverage según el estado estimado.
\end{itemize}

\subsection{Alcance del modelo}
El módulo incluye:
\begin{itemize}
  \item Entrenamiento de un HMM gaussiano sobre \texttt{regime_features} mediante \emph{Baum‑Welch} (EM) con regularización.
  \item Inferencia de probabilidades filtradas \(\mathbb{P}(S_t=k \mid \mathbf{X}_{1:t})\), suavizadas \(\mathbb{P}(S_t=k \mid \mathbf{X}_{1:T})\), y camino de estados vía \emph{Viterbi}.
  \item Etiquetado semántico de estados basado en estadísticas de \(\{\boldsymbol{\mu}_k, \boldsymbol{\Sigma}_k\}\) (volatilidad, liquidez, tendencia, drawdown proxies).
  \item Evaluación de estabilidad y persistencia de regímenes (duración media, estabilidad de \(\mathbf{A}\)).
\end{itemize}
No incluye:
\begin{itemize}
  \item Aplicación directa de una política de trading por estado (implementada en \texttt{regime_gating.py}).
  \item Generación de \texttt{regime_features} (modularidad explícita hacia \texttt{regime_features.py}).
\end{itemize}

\subsection{Dependencias aguas arriba/abajo}
Upstream:
\begin{itemize}
  \item \texttt{regime_features} outputs (matriz de features \(\mathbf{X}_{1:T} \in \mathbb{R}^{T \times d}\)).
  \item Calendario temporal y \texttt{train_window_mode} (expanding/rolling).
  \item Config HMM: \texttt{n\_states}, \texttt{covariance\_type}, \texttt{train\_window\_mode}, \texttt{min\_train\_length}, \texttt{regularization}, \texttt{random\_seed}, \texttt{state\_labeling\_policy}.
\end{itemize}
Downstream:
\begin{itemize}
  \item \texttt{models/regimes/regime_gating.py}: utiliza \(\mathbb{P}(S_t=k)\) para ajustar exposición.
  \item \texttt{validation/walkforward\_bias.py}: analiza look‑ahead y estabilidad de regímenes.
  \item \texttt{research/alpha\_discovery/regime\_detector.py}: investiga regímenes predictivos para alphas multi‑factor.
\end{itemize}

\begin{itemize}

  \item Salidas:
    \begin{itemize}

      \item \texttt{hmm\_params.json}: \(\boldsymbol{\pi}\), \(\mathbf{A}\), \(\{\boldsymbol{\mu}_k, \boldsymbol{\Sigma}_k\}\).
      \item \texttt{hmm\_summary.json}: métricas de ajuste y validación.
      \item \texttt{manifest.json}: metadatos de \texttt{run\_id}, ventana, features, etc.
    \end{itemize}
  \item Campos mínimos por fecha:
    \begin{itemize}
      \item \texttt{date}.
      \item \texttt{state\_id}.
      \item \texttt{p\_state\_0}…\texttt{p\_state\_k}.
      \item \texttt{state\_confidence} \(= \max_k \mathbb{P}(S_t=k)\).
      \item \texttt{run\_id}.
    \end{itemize}
\end{itemize}

\subsection{Metodología base: HMM gaussiano y Baum‑Welch}
Sea \(\mathcal{D} = \{\mathbf{x}_1,\dots,\mathbf{x}_T\}\) el conjunto de datos de entrenamiento. El likelihood del HMM gaussiano se define como:
\begin{equation}
\mathcal{L}(\boldsymbol{\theta} \mid \mathcal{D}) = \sum_{\mathbf{s}_{1:T}} \pi_{s_1} \prod_{t=2}^T A_{s_{t-1} s_t} \prod_{t=1}^T \mathcal{N}(\mathbf{x}_t \mid \boldsymbol{\mu}_{s_t}, \boldsymbol{\Sigma}_{s_t}),
\end{equation}
donde \(\boldsymbol{\theta} = (\boldsymbol{\pi}, \mathbf{A}, \{\boldsymbol{\mu}_k, \boldsymbol{\Sigma}_k\})\). El \textbf{Baum‑Welch} (EM) calcula:
\begin{equation}
\gamma_t(k) \propto \mathbb{P}(S_t=k \mid \mathcal{D}) = \frac{\alpha_t(k)\beta_t(k)}{\sum_{k'} \alpha_t(k')\beta_t(k')},
\end{equation}
\begin{equation}
\xi_t(i,j) \propto \mathbb{P}(S_{t-1}=i, S_{t}=j \mid \mathcal{D}) = \frac{\alpha_{t-1}(i) A_{ij} \mathcal{N}(\mathbf{x}_t \mid \boldsymbol{\mu}_j, \boldsymbol{\Sigma}_j)\beta_t(j)}{\sum_{i,j} \alpha_{t-1}(i) A_{ij} \mathcal{N}(\mathbf{x}_t \mid \boldsymbol{\mu}_j, \boldsymbol{\Sigma}_j)\beta_t(j)},
\end{equation}
y actualiza:
\begin{align}
\hat{\pi}_k & = \gamma_1(k), \\
\hat{A}_{ij} & = \frac{\sum_{t=2}^T \xi_t(i,j)}{\sum_{t=2}^T \gamma_{t-1}(i)}, \\
\hat{\boldsymbol{\mu}}_k & = \frac{\sum_{t=1}^T \gamma_t(k)\mathbf{x}_t}{\sum_{t=1}^T \gamma_t(k)}, \\
\hat{\boldsymbol{\Sigma}}_k & = \frac{\sum_{t=1}^T \gamma_t(k)(\mathbf{x}_t - \hat{\boldsymbol{\mu}}_k)(\mathbf{x}_t - \hat{\boldsymbol{\mu}}_k)^\top}{\sum_{t=1}^T \gamma_t(k)}.
\end{align}
Para \texttt{covariance\_type} restringido (tied, diagonal, spherical) se añade una regularización tipo Tikhonov o \(\ell_2\) para evitar colapsos de estados y mejora robustez en muestras finitas \cite{Rabiner1989, Zucchini2016}.

\subsection{Inferencia: filtrado, suavizado y Viterbi}
\begin{itemize}
  \item \textbf{Filtrado} (online): \(\mathbb{P}(S_t=k \mid \mathbf{X}_{1:t})\) usando el algoritmo \texttt{forward}:
    \begin{equation}
    \alpha_t(k) = \sum_{i=0}^{K-1} \alpha_{t-1}(i) A_{ik} \mathcal{N}(\mathbf{x}_t \mid \boldsymbol{\mu}_k, \boldsymbol{\Sigma}_k),
    \end{equation}
    \(\gamma_t(k) \propto \alpha_t(k)\). Útil para decisiones de trading en \texttt{t} real sin mirar al futuro.
  \item \textbf{Suavizado} (offline analítico): \(\mathbb{P}(S_t=k \mid \mathbf{X}_{1:T})\) con \texttt{forward–backward}:
    \begin{equation}
    \beta_t(k) = \sum_{j=0}^{K-1} A_{kj} \mathcal{N}(\mathbf{x}_{t+1} \mid \boldsymbol{\mu}_j, \boldsymbol{\Sigma}_j)\beta_{t+1}(j),
    \end{equation}
    \(\gamma_t(k) \propto \alpha_t(k)\beta_t(k)\). Validado solo para análisis y debugging, no para política de trading.
  \item \textbf{Viterbi path} (opcional): camino \(\hat{S}_{1:T}\) que maximiza \(\mathbb{P}(S_{1:T} \mid \mathbf{X}_{1:T})\), resuelto mediante programación dinámica \cite{Durbin1998}.
\end{itemize}

\subsection{Etiquetado semántico de estados}
Tras entrenamiento, cada estado \(k\) se caracteriza por:
\begin{itemize}
  \item \(\boldsymbol{\mu}_k\): niveles medios de \emph{volatilidad, liquidez, tendencia, drawdown proxies}.
  \item \(\boldsymbol{\Sigma}_k\): volatilidad condicional y correlaciones cruzadas.
\end{itemize}
Un policy de \texttt{state\_labeling} puede mapear:
\begin{itemize}
  \item \textbf{Risk‑on}: \(\mathbb{E}[R \mid k] > 0\), baja volatilidad, alta liquidez, tendencia positiva.
  \item \textbf{Risk‑off}: \(\mathbb{E}[R \mid k] < 0\), alta volatilidad, baja liquidez, tendencia negativa.
  \item \textbf{Crisis}: volatilidad muy alta, correlaciones de liquidez “flight‑to‑quality”.
  \item \textbf{Transitional/neutral}: régimen de transición, niveles intermedios.
\end{itemize}
La asignación automática mediante clustering o reglas de umbrales (e.g. \texttt{vol\_thresholds}, \texttt{trend\_thresholds}) se documenta en \texttt{hmm\_summary.json} y \texttt{state\_id} \(\to\) \texttt{regime\_label} persistido en \texttt{hmm\_params.json} \cite{Ang2012, Hamilton2010}.

\subsection{Reglas PIT y anti‑leakage}
\begin{itemize}
  \item En modo \emph{online} o \emph{walk‑forward} solo se usan \(\mathbb{P}(S_t=k \mid \mathbf{X}_{1:t})\). Nunca \(\mathbb{P}(S_t=k \mid \mathbf{X}_{1:T})\) con \(t < T\) conocido.
  \item Si se usa \emph{smoothing} para análisis, se etiqueta claramente como \texttt{offline} / \texttt{lookahead} en \texttt{hmm\_summary.json} y metadatos.
  \item En \texttt{config} se fuerza \texttt{inference\_mode} \(\in \{\text{filtered}, \text{smoothed}\}\) con flag de producción \texttt{prohibit\_smoothing\_in\_live}.
\end{itemize}

\subsection{Flujo del proceso (alto nivel)}
\begin{enumerate}
  \item \texttt{load\_hmm\_config} y \texttt{load\_regime\_features} construyen \(\mathcal{D} = \{\mathbf{x}_1,\dots,\mathbf{x}_T\}\).
  \item División \texttt{train\_window\_mode} (expanding/rolling) con \texttt{min\_train\_length}.
  \item \texttt{train\_hmm\_model} calibra \(\boldsymbol{\theta}\) mediante Baum‑Welch.
  \item \texttt{infer\_state\_probabilities} computa \(\gamma_t(k)\) y (opcional) Viterbi.
  \item \texttt{label\_states\_semantically} asigna \texttt{regime\_label} estructurado.
  \item \texttt{evaluate\_state\_stability} comprueba persistencia y estabilidad de \(\mathbf{A}\).
  \item \texttt{persist\_hmm\_outputs} serializa a \texttt{parquet} y \texttt{json} versiónados.
\end{enumerate}

\subsection{Métricas recomendadas}
\begin{itemize}
  \item \textbf{Log‑likelihood} train / valid, \textbf{AIC/BIC} para seleccionador de \texttt{n\_states} \cite{Bozdogan1987}:
    \begin{equation}
    \mathrm{BIC} = -2 \log \mathcal{L} + \log T \cdot \dim(\boldsymbol{\theta}).
    \end{equation}
  \item \textbf{State persistence}: duración media por estado \(\mathbb{E}[\text{chain\_length} \mid S_t=k]\).
  \item \textbf{Transition matrix stability}: distancia (e.g. \(\|\mathbf{A}_i - \mathbf{A}_j\|_F\)) entre ventanas rolling.
  \item \textbf{Confidence promedio}: \(\frac{1}{T} \sum_{t=1}^T \max_k \gamma_t(k)\).
\end{itemize}

\subsection{Validaciones mínimas y casos especiales}
\begin{itemize}
  \item \(\sum_{k=0}^{K-1} \gamma_t(k) = 1 \pm \varepsilon\) (tolerancia numérica).
  \item \texttt{n\_states\_effective} \(> 1\): detecta colapso a un solo estado dominante.
  \item Transiciones plausibles: \(\max_{i} \max_{j} A_{ij} \leq \rho < 1\) (evita absorción completa).
  \item Cobertura completa de fechas sin \texttt{NaN} en \texttt{state\_probs.parquet}.
  \item \textbf{Casos especiales}:
    \begin{itemize}
      \item \emph{Colapso de estados}: relanzar con \texttt{random\_seed} diferente o aumentar regularización.
      \item \emph{Features altamente correlacionadas}: opcional \texttt{PCA}/whitening antes de HMM.
      \item \emph{Cambios estructurales fu
