\section{Módulo \texttt{models/inference/explain.py}}

\subsection{Objetivo}
El módulo \texttt{models.inference.explain.py} tiene como objetivo principal \textbf{generar explicaciones de predicciones} para modelos lineales, GBDT y ensamblados (ensembles), con capacidad de:
\begin{itemize}
    \item Explicabilidad \textbf{global} (importancias globales, ranking de features).
    \item Explicabilidad \textbf{local} por predicción (contribuciones por feature).
    \item Soporte para métodos basados en \texttt{SHAP}, \texttt{permutation importance} y \texttt{coeficientes} según el tipo de modelo.
    \item Generación de reportes versionados por \texttt{run} para auditoría, debugging y confianza operativa.
\end{itemize}

Este sistema se enmarca dentro de la \textbf{reproducibilidad operativa} de modelos de trading, donde la trazabilidad de contribuciones por activo, fecha y feature es crítica para la toma de decisiones y la validación de estrategias de \textit{swing trading}.

\subsection{Contexto de XAI en Trading Cuantitativo}
En el contexto de la investigación cuantitativa industrial, la \textbf{Explainable AI (XAI)} juega un papel central en la apertura de modelos de predicción de precios y factores de riesgo. En particular, métodos como \texttt{SHAP} \cite{lundberg2017unified}, \texttt{permutation importance} \cite{breiman2001random} y análisis de coeficientes para modelos lineales \cite{james2013introduction} se han convertido en estándares de facto para:
\begin{itemize}
    \item Validar la consistencia de las señales generadas.
    \item Detectar overfitting o sesgos de features.
    \item Auditar la exposición a factores de riesgo en portfolios de alta rotación.
\end{itemize}

La combinación de explicaciones globales y locales permite pasar de una visión descriptiva (`feature X es importante`) a una visión causal aproximada (`feature X contribuye aproximadamente en \(\Delta\) puntos a la predicción de retorno en el activo \(s\) en el instante \(t\)`).

\subsection{Alcance Funcional}

\subsubsection{Incluye}
\begin{itemize}
    \item \textbf{Explicabilidad global}:
    \begin{itemize}
        \item Importancias globales de features (e.g.\ media de valores absolutos de SHAP, o medida de importancia por permutación).
        \item Ranking de features por relevancia predictiva.
    \end{itemize}
    \item \textbf{Explicabilidad local por predicción}:
    \begin{itemize}
        \item Descomposición de la predicción \( \hat{y}(x) \) en contribuciones por feature:
        \[
        \hat{y}(x) \approx \phi_0 + \sum_{j=1}^{p} \phi_j(x),
        \]
        donde \(\phi_j(x)\) es el valor de contribución del feature \(j\) para la instancia \(x\).
        \item Posibilidad de muestrear o explicar todo el conjunto de inferencia.
    \end{itemize}
    \item \textbf{Soporte por tipo de modelo}:
    \begin{itemize}
        \item Modelos lineales: explicación basada en coeficientes.
        \item GBDT (LightGBM, XGBoost, CatBoost): uso preferente de \texttt{TreeSHAP} \cite{lundberg2019efficient}.
        \item Ensembles: explicación por componente base más combinación final.
    \end{itemize}
    \item \textbf{Reportes versionados por \texttt{run}}:
    \begin{itemize}
        \item Metadata de experimento (configuración, fecha, seed, versión de librerías).
        \item Persistencia de explicaciones en formato estructurado (Parquet + JSON).
    \end{itemize}
\end{itemize}

\subsubsection{No incluye}
\begin{itemize}
    \item \textbf{Entrenamiento de modelos}: el módulo asume disponibilidad de modelos ya registrados.
    \item \textbf{Decisión final de cartera}: no se realiza optimización de pesos o selección de tickers; la decisión se confía a módulos superiores de \texttt{validation} y \texttt{portfolio}.
\end{itemize}

\subsection{Dependencias}

\subsubsection{Upstream}
\begin{itemize}
    \item \textbf{Modelos registrados}: modelos serializados (p.ej.\ LightGBM, XGBoost) identificados por \texttt{model_ref}.
    \item \textbf{Predicciones y features de inferencia}: conjunto de datos con:
    \begin{itemize}
        \item Variables \(\mathbf{X} \in \mathbb{R}^{n \times p}\) (features).
        \item Predicciones \(\mathbf{\hat{y}} \in \mathbb{R}^{n}\).
    \end{itemize}
    \item \textbf{Metadata de feature dictionary}: definición de grupo, categoría, horizonte temporal y unidad de cada feature.
\end{itemize}

\subsubsection{Downstream}
\begin{itemize}
    \item \textbf{Research / research pipeline evidence}: explicaciones globales y locales se consumen para validar hipótesis de factores predictivos.
    \item \textbf{Validation reports}: coherencia de señales, robustez de features y estabilidad de contribuciones.
    \item \textbf{Monitoring de comportamiento de modelo}: detección de drifts de interpretación (\textit{concept drift} en la importancia de features).
\end{itemize}

\subsection{Contrato de Entrada}

\subsubsection{Entradas mínimas}
\begin{itemize}
    \item \texttt{model_ref}: referencia del modelo (e.g.\ \texttt{lgbm_main_v3}).
    \item \texttt{features_path}: ruta a los features de inferencia (parquet o similar).
    \item \texttt{predictions_path}: ruta a las predicciones (scores parquet).
    \item \texttt{explain_config_path}: ruta a \texttt{configs/inference/explain.yaml}.
    \item \texttt{run_id}: identificador único del experimento de explicabilidad.
\end{itemize}

\subsubsection{Parámetros clave}
\begin{itemize}
    \item \texttt{explain_mode}: \texttt{global}, \texttt{local} o \texttt{both}.
    \item \texttt{method}: \texttt{shap}, \texttt{permutation}, \texttt{coef} según el modelo.
    \item \texttt{sample_size_local}: tamaño de muestra para explicaciones locales (si aplicable).
    \item \texttt{top_k_features}: número de features a mostrar en resúmenes.
\end{itemize}

\subsection{Contrato de Salida}

Los artefactos generados se persisten en:
\begin{center}
    \texttt{models/explain/\{run\_id\}/}
\end{center}

\subsubsection{Ficheros generados}
\begin{itemize}
    \item \texttt{global\_importance.parquet}:
    \begin{itemize}
        \item \textbf{Global feature importance} (por ejemplo, \(\bar{\phi}_j = \frac{1}{n} \sum_{i=1}^{n} |\phi_j(x_i)|\) para SHAP).
        \item Ranking de features por importancia.
    \end{itemize}
    \item \texttt{local\_explanations.parquet}:
    \begin{itemize}
        \item \textbf{Local feature attributions}:
        \[
        \phi_j(x_i) \quad \text{para cada instancia } x_i \text{ y feature } j.
        \]
    \end{itemize}
    \item \texttt{explain\_summary.json}:
    \begin{itemize}
        \item Estadísticas agregadas: número de observaciones, porcentaje de varianza explicada, cobertura de features.
    \end{itemize}
    \item \texttt{manifest.json}:
    \begin{itemize}
        \item Metadatos de run: seed, versión de librerías, configuración, fecha.
    \end{itemize}
\end{itemize}

\subsubsection{Campos mínimos \texttt{local\_explanations}}
\begin{itemize}
    \item \texttt{date}: fecha de la operación.
    \item \texttt{symbol}: ticker.
    \item \texttt{model_id}: identificador del modelo.
    \item \texttt{y\_pred}: predicción.
    \item \texttt{feature\_name}: nombre del feature.
    \item \texttt{contribution}: \(\phi_j(x)\).
    \item \texttt{rank\_abs\_contribution}: ranking de la contribución absoluta.
\end{itemize}

\subsection{Metodologías Implementadas}

\subsubsection{Modelos lineales}
Para modelos lineales de la forma:
\[
\hat{y}(x) = \beta_0 + \sum_{j=1}^{p} \beta_j x_j,
\]
el módulo usa la descomposición directa:
\[
\phi_j(x) = \beta_j x_j.
\]
Esta aproximación es exacta y computacionalmente eficiente, y se basa en la linealidad de la función de predicción. La interpretación es inmediata:
\begin{itemize}
    \item El signo de \(\beta_j\) indica la dirección de la relación con el retorno predicho.
    \item La magnitud de \(\beta_j x_j\) mide la contribución de la combinación feature–valor al retorno.
\end{itemize}

\subsubsection{GBDT (TreeSHAP y permutación)}
Para modelos tipo GBDT, se utilizan dos métodos:
\begin{itemize}
    \item \textbf{TreeSHAP} \cite{lundberg2019efficient}:
    \begin{itemize}
        \item Versión eficiente de SHAP para modelos de árboles.
        \item Calcula valores de Shapley:
        \[
        \phi_j(x) = \sum_{S \subseteq F \setminus \{j\}} \frac{|S|!(p-|S|-1)!}{p!} \left[ f(x_{S \cup \{j\}}) - f(x_S) \right],
        \]
        donde \(F\) es el conjunto de features y \(f\) es el modelo de árboles.
        \item TreeSHAP reduce la complejidad de la suma sobre coaliciones empleando la estructura de los árboles: \(\mathcal{O}(TLD^2)\), donde \(T\) es el número de árboles, \(L\) las hojas y \(D\) la profundidad.
    \end{itemize}
    \item \textbf{Permutation feature importance}:
    \begin{itemize}
        \item Como \textit{fallback} cuando SHAP no es aplicable o demasiado costoso.
        \item Se define la importancia de la feature \(j\) como:
        \[
        \text{PFI}_j = \mathbb{E}[L(\mathbf{y}, \hat{\mathbf{y}})] - \mathbb{E}[L(\mathbf{y}, \hat{\mathbf{y}}_{j^\text{perm})],
        \]
        donde \(L\) es la función de pérdida, \(\hat{\mathbf{y}}\) la predicción original y \(\hat{\mathbf{y}}_{j^\text{perm}}\) la predicción con la feature \(j\) permutada.
        \item Este método es el estándar de la industria para la validación de importancias globales en modelos de ensamblado \cite{breiman2001random}.
    \end{itemize}
\end{itemize}

\subsubsection{Ensembles}
En el caso de modelos de tipo ensemble (e.g.\ voting, stacking, blending), el módulo implementa:
\begin{itemize}
    \item Explicación por componente base:
    \[
    \phi_j^{(b)}(x) \quad \text{para cada modelo base } b.
    \]
    \item Explicación de la combinación final:
    \[
    \phi_j(x) = \sum_{b=1}^{B} w_b \phi_j^{(b)}(x),
    \]
    donde \(w_b\) son los pesos de agregación (e.g.\ constantes o dependientes de la instancia).
\end{itemize}
Este enfoque permite auditar la contribución de cada modelo base al comportamiento final del ensemble, lo cual es clave en entornos de trading donde la robustez de las señales es crítica.

\subsection{Reglas de Reproducibilidad}

\begin{itemize}
    \item Fijar una \texttt{seed} para el muestreo de instancias locales:
    \[
    \text{seed} = \text{run\_id} \oplus \text{timestamp},
    \]
    almacenada en el \texttt{manifest.json}.
    \item Registrar la versión de las librerías de explicabilidad (e.g.\ \texttt{shap}, \texttt{interpret}, \texttt{sklearn}).
    \item Guardar el \textbf{subset} utilizado para explicaciones locales (o sus índices de muestreo) para poder reproducir exactamente las contribuciones.
\end{itemize}

Estas reglas se alinean con las recomendaciones de \textit{reproducible machine learning} \cite{veit2023reproducibility} en contextos de alta frecuencia donde la trazabilidad de decisiones es regulatoria y operativa.

\subsection{Flujo de Ejecución}

\begin{enumerate}
    \item \textbf{Cargar modelo, features y predicciones}:
    \begin{itemize}
        \item Cargar el modelo registrado \texttt{model_ref}.
        \item Cargar \(\mathbf{X}\) y \(\mathbf{\hat{y}}\) desde \texttt{features\_path} y \texttt{predictions\_path}.
    \end{itemize}
    \item \textbf{Resolver método de explicación}:
    \begin{itemize}
        \item Clasificar tipo de modelo y seleccionar \texttt{method} (coef, SHAP, permutación).
        \item Derivar parámetros de ejecución (e.g.\ número de permutaciones, tamaño de batch).
    \end{itemize}
    \item \textbf{Calcular importancias globales}:
    \begin{itemize}
        \item Para SHAP:
        \[
        \bar{\phi}_j = \frac{1}{n} \sum_{i=1}^{n} |\phi_j(x_i)|.
        \]
        \item Para permutación:
        \[
        \text{PFI}_j = \text{mean\_loss\_original} - \text{mean\_loss\_permuted}_j.
        \]
    \end{itemize}
    \item \textbf{Calcular explicaciones locales}:
    \begin{itemize}
        \item Aplicar el método seleccionado a la muestra completa o a un subconjunto de tamaño \texttt{sample\_size\_local}.
    \end{itemize}
    \item \textbf{Persistir reportes}:
    \begin{itemize}
        \item Escribir \texttt{global\_importance.parquet}, \texttt{local\_explanations.parquet}, \texttt{explain\_summary.json} y \texttt{manifest.json}.
    \end{itemize}
\end{enumerate}

\subsection{Validaciones Implementadas}

\begin{itemize}
    \item \textbf{Contribuciones finitas}:
    \begin{itemize}
        \item Verificar que \(\phi_j(x)\) no sea \texttt{NaN} ni \texttt{Inf} para ninguna instancia.
    \end{itemize}
    \item \textbf{Consistencia de la suma de contribuciones}:
    \begin{itemize}
        \item Para métodos que cumplen el axioma de \textit{efficiency} (SHAP, permutación adaptada):
        \[
        y_{\text{pred}} \approx \phi_0 + \sum_{j=1}^{p} \phi_j(x),
        \]
        dentro de un umbral de tolerancia \(\epsilon\).
    \end{itemize}
    \item \textbf{Cobertura mínima de explicaciones locales}:
    \begin{itemize}
        \item Verificar que al menos un porcentaje \(\gamma\) de instancias tiene explicaciones locales calculadas.
    \end{itemize}
\end{itemize}

Estas validaciones garantizan que las explicaciones
