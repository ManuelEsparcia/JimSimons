\section{Introducción y Contexto Estratégico}
En swing trading, drifts (distribución features/labels) causan $\Delta \alpha < -5\%$/mes por regime change (e.g., bull$\to$bear). Este módulo compara ventanas rolling vs baseline, triggerando retrain si PSI>0.1, clave para Sharpe persistente >2 [web:47][web:49].

Integra como gate post-features en daily\_pipeline, governance como MLflow/Alibi Detect.

\section{Matemáticas Fundamentales del Drift}
\textbf{PSI (Population Stability Index)}: $PSI = \sum (p_{cur} - p_{ref}) \ln \frac{p_{cur}}{p_{ref}}$, bins=10, >0.1=drift fuerte.

\textbf{KS Statistic}: $KS = \sup |CDF_{cur}(x) - CDF_{ref}(x)|$, p-value via Lilliefors.

\textbf{Jensen-Shannon Divergence}: $JS(P||Q) = \frac{1}{2} KL(P||M) + \frac{1}{2} KL(Q||M)$, $M=(P+Q)/2$, simétrica [0,1].

\textbf{Calibration Drift (ECE)}: $ECE = \sum \frac{n_k}{n} |acc_k - conf_k|$, Brier: $\frac{1}{N} \sum (p_i - y_i)^2$.

\textbf{Performance Drift}: $\Delta IC = |IC_{rolling} - IC_{baseline}|$, $IC = corr(pred, target)$; $\Delta Sharpe = |S_{cur} - S_{ref}|$.

Global: $D_g = \max( w_f D_f + w_p D_p + w_{perf} D_{perf} )$, fail si $D_g > \theta$.

import alibi_detect  # O alibi-detect-cd
def main(config, feat_cur, feat_ref, pred_cur, perf_cur, run_id):
    cfg = load_config(config)
    # Features Drift
    drifts_f = {}
    for col in feat_cur.columns:
        psi = psi_stat(feat_cur[col], feat_ref[col])
        ks = ks_2samp(feat_cur[col], feat_ref[col])
        drifts_f[col] = {'PSI': psi, 'KS': ks.stat}
    
    # Preds/Model Drift
    ece_delta = ece(pred_cur['score'], labels_cur) - ece_ref
    ic_delta = np.corrcoef(pred_cur['score'], perf_cur['target'])[0,1] - ic_ref
    
    # Aggregate
    D_global = np.max([np.mean(list(drifts_f.values())), abs(ece_delta), abs(ic_delta)])
    status = 'fail' if D_global > cfg.drift_threshold else 'pass'
    actions = recommend_action(D_global)  # 'retrain' if >0.25
    
    persist_reports({'drifts': drifts_f, 'global': D_global, 'action': actions}, run_id)
    return status
\end{lstlisting}
