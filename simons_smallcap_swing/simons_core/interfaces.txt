\section{Objetivo y Paradigma Simons}
Define protocolos ABC/Pydantic para estandarizar DataProvider $\to$ ModelTrainer $\to$ PortfolioEngine en swing trading. Asegura reproducibilidad (seed + config_hash), clave para Renaissance (66\% retornos via pipelines modulares) \cite{simons_returns}. Versionado SemVer previene drifts.

\section{Alcance y Interfaces Nucleares}
\begin{itemize}
\item \texttt{DataProvider}: \texttt{fetch(asof\_date)}.
\item \texttt{FeatureBuilder}: $\mathbf{X} = f(\mathbf{Y}, \tau)$.
\item \texttt{ModelTrainer}: $\hat{\theta} = \arg\min L(\theta; D)$.
\item \texttt{PortfolioEngine}: $\mathbf{w}^* = \arg\max \mu^T \mathbf{w} - \lambda \mathbf{w}^T \Sigma \mathbf{w}$.
\item \texttt{Validator}: Gate con $\mathbb{P}(\text{fail}) < \epsilon=10^{-3}$.
\end{itemize}

Tipos: \texttt{RunContext(seed=42)}, \texttt{GateResult(score \in [0,1])}.

\section{Matemáticas Formales de Contratos}
\subsection{Tipos Dependientes y Validación}
Contrato: $\forall impl \in Interface, \, \Sigma(impl.in) = impl.out$ dimensionalmente.
Probabilistic gate:
\[
\text{pass} \iff \prod p_i(\text{metric}_i | \theta) > \tau, \quad \tau=0.95
\]
Bayesian validation \cite{lopez_ml}.

\subsection{Optimización Portfolio}
Quadratic programming en \texttt{PortfolioEngine}:
\[
\mathbf{w}^* = \min_{\mathbf{w}} \frac{1}{2} \mathbf{w}^T \Sigma \mathbf{w} - \lambda \mu^T \mathbf{w}, \quad \mathbf{1}^T \mathbf{w}=1, \, w_i \geq 0
\]
Swing-adapted: Constraints long-only, turnover $<20\%$/rebalance \cite{chan}.
