\section{Objetivo Matemático}
Proveer I/O columnar optimizado para datasets trading:
\begin{equation}
\text{I/O Saved} = S_{\text{raw}} \cdot (1 - CR) \cdot (1 - \text{Selectivity})
\end{equation}
donde $CR \approx 10x$ compression ratio Parquet, Selectivity $\approx 0.08$ monthly filters.

\section{Alcance Production-Grade}
\begin{itemize}
    \item \textbf{write\_parquet()} versioned snapshots con manifest
    \item \textbf{read\_parquet()} column projection + pushdown filters
    \item Schema contracts + checksum validation
    \item Partitioning: date/symbol (pruning $\approx 92\%$ I/O)
\end{itemize}

\section{API Industrial}
\begin{lstlisting}
def write_parquet(df: pd.DataFrame, 
                  path: str, 
                  schema_ref: str,
                  partition_cols: List[str] = ['date', 'symbol'],
                  write_mode: str = 'fail_if_exists') -> Dict:
    """
    Atomic write con schema validation + manifest
    Returns: {'row_count': N, 'schema_hash': SHA256, 'checksum': MD5}
    """
    manifest = generate_manifest(df, partition_cols)
    pq.write_to_dataset(df, path, ...)
    return manifest
\end{lstlisting}

\section{Reglas de Escritura Robustas}
\begin{enumerate}
    \item \textbf{Schema Pre-validation}: PyArrow Schema.equals()
    \item \textbf{Deduplication}: PK hash collisions $P < 10^{-9}$
    \item \textbf{Write Modes}:
    \begin{itemize}
        \item \texttt{fail\_if\_exists}: Zero-downtime validation
        \item \texttt{overwrite\_partition}: Date-granular updates
        \item \texttt{append\_versioned}: Audit trail completo
    \end{itemize}
\end{enumerate}

\section{Optimización Lectura Columnar}
\begin{lstlisting}

df = read_parquet(

\section{Manejo de Errores Production}
\begin{itemize}
    \item \textbf{Schema Mismatch} $\rightarrow$ \texttt{Raise SchemaValidationError}
    \item \textbf{Corrupt Partition} $\rightarrow$ \texttt{.isolate\_and\_log()}
    \item \textbf{Path Injection} $\rightarrow$ \texttt{PathValidator.reject()}
\end{itemize}
