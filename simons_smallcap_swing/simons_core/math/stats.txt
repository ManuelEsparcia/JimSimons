\section{Objetivo y Fundamentos Teóricos}
Utilidades estadísticas reproducibles para research, validación y reporting en swing trading. En finance, stats deben ser robustas a NaN y numéricamente estables (e.g., corr via SVD para singularidad).

\textbf{Métricas clave definidas:}
\begin{itemize}
\item Sharpe Ratio: \( SR = \frac{\mathbb{E}[r - r_f]}{\sqrt{\Var(r)}} \) [Sharpe, 1966].
\item Sortino: Penaliza downside vol \( \sigma_d = \sqrt{\mathbb{E}[\min(r - \tau, 0)^2]} \) [Sortino \& van Greis, 1998].
\item Calmar: \( \frac{\bar{r}}{\max DD} \).
\item Drawdown: \( DD_t = 1 - \frac{P_t}{\max_{s \leq t} P_s} \).
\end{itemize}

\textbf{Referencias:}
\begin{itemize}
\item Libro: \textit{Quantitative Trading} (Chan, 2020) -- Backtest metrics.
\item Paper: ``Deflated Sharpe Ratio'' (Bailey \& López de Prado, 2014) -- Multiple testing.
\item SOTA: Probabilistic Sharpe via Bayesian [Kozak et al., 2020].
\end{itemize}

\section{Alcance y Funciones Core}

\subsection{Momentos Básicos: mean/std/var}
Wrappers con ddof=1 (sample), NaN-safe:
\[
\bar{x} = \frac{1}{n} \sum x_i, \quad s^2 = \frac{1}{n-1} \sum (x_i - \bar{x})^2.
\]

\subsection{Rolling Stats}
Vectorizadas Pandas:
- \texttt{rolling\_mean(x, window=20)}: Media móvil.
- \texttt{rolling\_zscore(x, window)}: \( z_t = \frac{x_t - \bar{x}_{t-w:t}}{\sigma_{t-w:t}} \).

\subsection{Correlaciones: spearman\_ic/pearson\_corr}
Spearman: Rank corr, non-paramétrico (IC en alpha research).
\[
\rho = 1 - \frac{6 \sum d_i^2}{n(n^2 - 1)}, \quad IC = |\corr(\rank(r), \rank(pred))|.
\]

\subsection{winsorize(x, limits=(0.01, 0.99))}
Clamp percentiles para mitigar outliers.

\subsection{Métricas Financieras}
\begin{align}
\text{Sharpe}(r, r_f=0) &= \frac{\mean(r) - r_f}{\std(r)} \sqrt{252}, \\
\text{Sortino}(r, \tau=0) &= \frac{\mean(r - \tau)}{\std((r - \tau)_-)} \sqrt{252}, \\
\text{Calmar}(r) &= \frac{\mean(r)}{\max(DD)} \cdot 252.
\end{align}
Drawdown series: Cumulative max troughs.

\textbf{Código vectorizado pro:}
\begin{lstlisting}[language=Python, basicstyle=\ttfamily\small]
import numpy as np
import pandas as pd
from typing import Union

def sharpe_ratio(returns: np.ndarray, rf: float = 0.0, periods: int = 252) -> float:
    excess = returns - rf
    return np.mean(excess) / np.std(excess) * np.sqrt(periods)

def max_drawdown(returns: np.ndarray) -> float:
    cumprod = np.cumprod(1 + returns)
    peak = np.maximum.accumulate(cumprod)
    dd = (cumprod - peak) / peak
    return np.min(dd)

def rolling_zscore(x: pd.Series, window: int) -> pd.Series:
    mean = x.rolling(window).mean()
    std = x.rolling(window).std()
    return (x - mean) / std

def spearman_ic(pred: np.ndarray, actual: np.ndarray) -> float:
    return np.corrcoef(np.argsort(pred), np.argsort(actual))[0,1]
\end{lstlisting}

\section{Reglas Numéricas y API}
- NaN: Ignorar via \texttt{np.nanmean}, propagar si all-NaN.
- Estabilidad: Clip eps $10^{-12}$, corr via \texttt{np.linalg.svd} para $\Sigma$ singular.
- Tipos: Union[np.ndarray, pd.Series].
- Convenciones: Annualize $\sqrt{252}$, rf=0 default.

\section{Validaciones Exhaustivas}
- Casos: Normal, Cauchy (fat tails).
- Invariantes: std $\geq 0$, |corr| $\leq 1$.
- CLI: \texttt{python -m simons\_core.math.stats --self-test}.

Ejemplo:
\begin{lstlisting}[language=Python]
r = np.random.normal(0.001, 0.02, 1000)
print(sharpe_ratio(r))  # ~0.8 annual
print(max_drawdown(r))  # ~-0.15
