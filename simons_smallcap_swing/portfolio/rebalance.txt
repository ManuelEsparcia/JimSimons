\section{Introducción}

El módulo \texttt{portfolio/rebalance.py} traduce pesos objetivo post-constraints en órdenes de rebalanceo ejecutables, aplicando políticas que minimizan turnover innecesario, respetan budgets operativos y generan planes accionables para swing trading en small caps.

En swing trading, rebalanceos frecuentes (semanal/mensual) pueden erosionar alpha por costes; este módulo controla turnover total, aplica thresholds y filtra micro-trades, integrando lot sizing para viabilidad. Produce órdenes con notional, shares y motivos de skip.

Basado en \citet{kissell2006optimal} (optimal trade scheduling) y \citet{lprado2018advances} (turnover control en backtests).

\section{Formulación Matemática}

\subsection{Definiciones}
$\mathbf{w}^{\mathrm{target}} \in \mathbb{R}^N$: Pesos objetivo.
$\mathbf{w}^{\mathrm{current}} \in \mathbb{R}^N$: Pesos actuales (positions / NAV).
$\mathbf{p} \in \mathbb{R}^N_+$: Precios de referencia.

\textbf{Delta Pesos}:
\[
\Delta w_i = w_i^{\mathrm{target}} - w_i^{\mathrm{current}}
\]

\textbf{Shares Objetivo y Delta}:
\[
s_i^{\mathrm{target}} = \frac{w_i^{\mathrm{target}} \cdot \mathrm{NAV}}{\mathrm{p}_i}, \quad \Delta s_i = s_i^{\mathrm{target}} - s_i^{\mathrm{current}}
\]
\[
\mathrm{Notional}_i = |\Delta s_i| \cdot p_i
\]

\textbf{Turnover Planeado}:
\[
\mathrm{Turnover} = \sum_i |\Delta w_i|, \quad \mathrm{TotalNotional} = \sum_i |\mathrm{Notional}_i|
\]

\subsection{Políticas de Rebalanceo}

\textbf{Threshold Policy}:
\[
\mathrm{Flag}_i = \mathbb{I}(|\Delta w_i| \geq \theta_w) \quad \mathrm{o} \quad \mathbb{I}(|\mathrm{Notional}_i| \geq n_{\min})
\]

\textbf{Budget Constraint}:
\[
\sum_{i \in \mathcal{A}} |\Delta w_i| \leq b_{\mathrm{turnover}}, \quad \text{priorizar por } |\Delta w_i| / \mathrm{cost}_i
\]

\textbf{Staggered (Multi-Period)}:
\[
\Delta s_{i,t} = \gamma_t \cdot \Delta s_i, \quad \sum_t \gamma_t = 1, \quad \gamma_t \sim \mathrm{Geom}(\rho)
\]

\textbf{Rounding Lots}:
\[
s_i^{\mathrm{rounded}} = \round\left( \frac{s_i^{\mathrm{target}}}{l_i} \right) \cdot l_i, \quad l_i = \mathrm{lot\_size}_i
\]
Residual tracked como cash adjustment.

\section{Implementación Algorítmica}

Flujo secuencial:

\begin{enumerate}
\item Cargar $\mathbf{w}^{\mathrm{target}}$, posiciones actuales, $\mathbf{p}$, config ($\theta_w, b_{\mathrm{turnover}}, l_i$).
\item Computar $\Delta \mathbf{w}$, $s^{\mathrm{target}}$, $\Delta \mathbf{s}$, $\mathrm{Notional}$.
\item Aplicar threshold: Filtrar $|\Delta w_i| < \theta_w$.
\item Budget turnover: Ordenar por $|\Delta w_i|$ descendente, acumular hasta $b$.
\item Rounding: Aplicar lot sizing conservador (hacia cero para longs/shorts).
\item Generar órdenes: Symbol, side (buy/sell), shares, notional estimado, flag/skip\_reason.
\item Validar: $\sum \mathrm{Notional} \leq$ budget, consistencia shares-precio.
\item Persistir parquet órdenes + JSON summary.
\end{enumerate}

\subsection{Validaciones}
- $\sum |\Delta w_i^{\mathrm{flag}}| \leq b + \epsilon$.
- $\forall i$: $|\Delta s_i \cdot p_i - \Delta w_i \cdot \mathrm{NAV}| < \delta$.
- No órdenes con notional < $n_{\min}$ salvo cierre.
- Simulación forward: Tracking error < 1\% post-rebalance.
