\section{Módulo: \texttt{features/fundamentals\_deltas.py}}

\paragraph{Objetivo general}
Este módulo construye \textbf{features de cambio fundamental} (QoQ, YoY, aceleraciones) a partir de la serie de \texttt{fundamentals\_core}, capturando dinámicas de盈利, calidad, liquidez de balance y estructura de capital, con enfoque en \textit{señales de inflexión} de small caps. El objetivo es alimentar estrategias de swing trading multi‑factor con sensibilidad a cambios de régimen contable, ampliando el rango de horizontes de momen\-tum y reversión.

\subsection{Marco teórico y notación}

Sea \( \mathcal{S} \) el conjunto de símbolos y \( \mathcal{T} \) el rango de fechas. Denotamos:
\begin{itemize}
    \item \( F_{t}^{(s)} \in \mathbb{R} \) una métrica fundamental core en fecha \( t \) y simbolo \( s \) (por ejemplo, \texttt{fund\_revenue}, \texttt{fund\_profit\_margin}, \texttt{fund\_leverage}).
    \item \( F_{t-l}^{(s)} \) su valor lagueado, con \texttt{asof\_date} correcto en el marco PIT.
\end{itemize}

La serie de deltas fundamentales se define como:
\[
\Delta F_{t}^{(s)} = F_{t}^{(s)} - F_{t-l}^{(s)},
\]
mientras que \textbf{aceleraciones} de cambios (segunda diferencias) se definen:
\[
\Delta^2 F_{t}^{(s)} = \Delta F_{t}^{(s)} - \Delta F_{t-l}^{(s)}.
\]

\subsection{Features de deltas y aceleraciones}

\subsubsection{Crecimiento de ingresos}
\begin{itemize}
    \item \texttt{fund\_revenue\_growth\_qoq}:
        \[
        \text{Rev\_G\_QoQ}_t^{(s)} = \frac{F_{t}^{(s,\text{revenue})} - F_{t-1q}^{(s,\text{revenue})}}{F_{t-1q}^{(s,\text{revenue})}},
        \]
        donde \( t-1q \) es el periodo contable inmediatamente anterior (QoQ) disponible en \texttt{asof\_date} \(\leq t\).
    \item \texttt{fund\_revenue\_growth\_YoY}:
        \[
        \text{Rev\_G\_YoY}_t^{(s)} = \frac{F_{t}^{(s,\text{revenue})} - F_{t-1y}^{(s,\text{revenue})}}{F_{t-1y}^{(s,\text{revenue})}},
        \]
        capturando crecimiento de largo plazo.
\end{itemize}

\subsubsection{Delta de margen}
\begin{itemize}
    \item \texttt{fund\_margin\_delta\_qoq}:
        \[
        \Delta \text{PM\_QoQ}_t^{(s)} = \text{PM}_{t}^{(s)} - \text{PM}_{t-1q}^{(s)},
        \]
        donde \texttt{PM} es \texttt{fund\_profit\_margin} (core).
\end{itemize}

\subsubsection{Delta de leverage}
\begin{itemize}
    \item \texttt{fund\_leverage\_delta\_yoy}:
        \[
        \Delta \text{Leverage\_YoY}_t^{(s)} = \text{Leverage}_{t}^{(s)} - \text{Leverage}_{t-1y}^{(s)},
        \]
        midiendo cambios de estructura de capital y riesgo de crédito.
\end{itemize}

\subsubsection{Delta de flujo de caja}
\begin{itemize}
    \item \texttt{fund\_cashflow\_delta\_qoq}:
        \[
        \Delta \text{CFO\_QoQ}_t^{(s)} = \text{CFO}_{t}^{(s)} - \text{CFO}_{t-1q}^{(s)},
        \]
        donde \texttt{CFO} es el cash flow operativo TTM, normalizado por activo o ingreso si procede.
\end{itemize}

\subsubsection{Aceleración de crecimiento}
\begin{itemize}
    \item \texttt{fund\_growth\_acceleration}:
        \[
        A_t^{(s)} = \Delta^2 \text{Rev\_G}_t^{(s)} = \Delta \text{Rev\_G}_t^{(s)} - \Delta \text{Rev\_G}_{t-l}^{(s)},
        \]
        donde \( \Delta \text{Rev\_G}_t^{(s)} \) es el delta de crecimiento anterior.
        La aceleración mide \textit{inflexiones} (cambio de concavidad) en la trayectoria de ingresos.
\end{itemize}

\subsection{Reglas PIT y manejo de lags}

\paragraph{Lags consistentes con asof\_date}
\begin{itemize}
    \item Solo se comparan observaciones \texttt{asof\_date} \(\leq t\) y \(\leq t-l\).
    \item Nunca se usan \texttt{period\_end} futuro no disponible en \texttt{asof\_date} actual.
\end{itemize}

\paragraph{Alineación temporal estricta}
\begin{itemize}
    \item Se construye una serie lagged \( F_{t-l}^{(s)} \) asegurando que su \texttt{asof\_date} no exceda \texttt{t-l}.
    \item Para cada periodo (QoQ, YoY), se usa el \texttt{asof} más reciente válido, no interpolación hacia el futuro.
\end{itemize}

\subsection{Flujo de procesamiento}

\paragraph{Alto nivel}
1. \texttt{load\_fundamentals\_core\_with\_lags(...)}:
    - Carga \texttt{features/fundamentals\_core/...} y alinea lags QoQ/YoY consistentes con \texttt{asof\_date}.
2. \texttt{compute\_deltas(...)}:
    - Calcula deltas absolutos y relativos:
        \[
        \Delta F_t^{(s)} = F_t^{(s)} - F_{t-l}^{(s)},
        \quad
        \Delta F_{t,\text{rel}}^{(s)} = \Delta F_t^{(s)} / F_{t-l}^{(s)}.
        \]
3. \texttt{compute\_accelerations(...)}:
    - Deriva aceleraciones como segunda diferencia temporal:
        \[
        \Delta^2 F_t^{(s)} = \Delta F_t^{(s)} - \Delta F_{t-l}^{(s)}.
        \]
4. \texttt{apply\_clipping\_normalization(...)}:
    - Aplica clipping a percentiles \([p_{\min}, p_{\max}]\) o winsorización.
    - Opcionalmente transforma en rank o zscore robusto por fecha para uso en \texttt{cross\_sectional.py}.
5. \texttt{persist\_fundamental\_deltas(...)}:
    - Persiste outputs en \texttt{features/fundamentals\_deltas/date=...} junto con resumen y manifest.

\subsection{Validaciones de sanidad y consistencia}

\paragraph{Coherencia temporal}
\begin{itemize}
    \item Verificar que \( t-l \in \mathcal{T} \) y \texttt{asof\_date} del lag \(\leq t-l\).
    \item Chequear que no haya \textit{leakage} en joins entre períodos futuros y pasados.
\end{itemize}

\paragraph{Rangos de deltas plausibles}
\begin{itemize}
    \item \texttt{growth\_qoq} y \texttt{growth\_yoy} en rangos razonables (por ejemplo, \([-300\%, 300\%]\) salvo para casos de quiebra/renacimiento).
    \item \texttt{delta\_margin} acotados por \([-100\%, 100\%]\) típicamente.
\end{itemize}

\paragraph{Cobertura mínima}
\begin{itemize}
    \item \texttt{coverage\_rate} de deltas por feature:
        \[
        \rho_{\text{cov}}^{(\Delta f)} =
        \frac{1}{|\mathcal{T}|} \sum_{t \in \mathcal{T}}
        \rho_{\text{cov}}^{(\Delta f,t)} \geq \texttt{min\_coverage\_pct}.
        \]
\end{itemize}

\subsection{Salidas y estructura de almacenamiento}

\paragraph{Outputs principales}
\begin{itemize}
    \item \texttt{features/fundamentals\_deltas/date=\{yyyy-mm-dd\}/part-*.parquet}:
        \begin{itemize}
            \item \texttt{date}, \texttt{symbol}, \texttt{run\_id}.
            \item Columnas de deltas:
                \begin{itemize}
                    \item \texttt{fund\_revenue\_growth\_qoq}, \texttt{fund\_revenue\_growth\_yoy},
                    \texttt{fund\_margin\_delta\_qoq}, \texttt{fund\_leverage\_delta\_yoy},
                    \texttt{fund\_cashflow\_delta\_qoq}, \texttt{fund\_growth\_acceleration}.
                \end{itemize}
        \end{itemize}
    \item \texttt{features/fundamentals\_deltas/summary\_\{run\_id\}.json}:
        \begin{itemize}
            \item \texttt{mean}, \texttt{p50}, \texttt{p99}, \texttt{coverage\_rate} por feature.
        \end{itemize}
    \item \texttt{features/fundamentals\_deltas/manifest\_\{run\_id\}.json}:
        \begin{itemize}
            \item \texttt{run\_id}, \texttt{config\_hash}, \texttt{fundamentals\_core\_hash}, \texttt{deltas\_config\_path}.
        \end{itemize}
\end{itemize}

\subsection{CLI y uso en el pipeline global}

\paragraph{CLI recomendada}
\begin{verbatim}
python -m features.fundamentals_deltas ^
  --fundamentals-core-path features/fundamentals_core ^
  --config-path configs/features/fundamentals_deltas.yaml ^
  --run-id fundamentals_deltas_20260305_001
\end{verbatim}

\paragraph{Uso downstream}
\begin{itemize}
    \item \texttt{features/build\_features.py}: integra \texttt{fund\_*_delta_*} y \texttt{fund\_growth\_acceleration} como factores de \textit{quality growth} y \textit{momentum de calidad}.
    \item \texttt{models/*/train\_*.py}: construye factores de reversión de calidad y crecimiento de earnings, siguiendo literatura de \textit{value–quality} y \textit{earnings momentum} (Novy‑Marx et al.). \cite{novy2013}
    \item \texttt{backtest/run\_walkforward.py}: explora portfolios long–short basados en aceleraciones de crecimiento y calidad de flujo de caja.
\end{itemize}

\subsection{Definition of Done}

\begin{itemize}
    \item Conjunto de \textbf{deltas fundamentales} PIT robustos, comparables entre fechas.
    \item Alineación temporal estricta (QoQ/YoY) sin leakage.
    \item Transformaciones opcionales de clipping/normalización compatibles con el paso de \texttt{cross\_sectional.py}.
\end{itemize}
