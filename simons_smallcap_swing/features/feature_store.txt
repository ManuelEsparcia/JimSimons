\section{Módulo: \texttt{features/feature\_store.py}}

\paragraph{Objetivo general}
Este módulo implementa un \textbf{feature store inmutable y versionado} para almacenar la matriz de señales PIT, garantizando que tanto el entrenamiento como la inferencia de modelos de swing trading consuman datos consistentes, reproducibles y bien tipados. El objetivo es separar la construcción de features de su consumo, manteniendo un contrato de lectura estable entre \texttt{build\_features.py} y \texttt{models/*/train\_*.py} o \texttt{backtest/run\_walkforward.py}.

\subsection{Marco teórico y notación}

Sea \( \mathcal{T} \) el conjunto de fechas de almacenamiento y \( \mathcal{S} \) el conjunto de símbolos de small caps observados en esas fechas. El \textbf{feature store} se define como un conjunto de snapshots:

\[
\mathcal{X}^{(\kappa)} = \big\{ X_{t}^{(\kappa)} \in \mathbb{R}^{|\mathcal{S}_t| \times F_\kappa} \big\}_{t \in \mathcal{T}_\kappa},
\]
donde:
\begin{itemize}
    \item \( \kappa \) indexa la versión o \texttt{run\_id} del store.
    \item \( F_\kappa \) es el conjunto de features disponibles en dicha versión.
    \item \( \mathcal{T}_\kappa \subseteq \mathcal{T} \) es el rango de fechas cubiertas.
\end{itemize}

El objetivo de \texttt{feature\_store.py} es:
\begin{itemize}
    \item Escribir snapshots inmutables \( \mathcal{X}^{(\kappa)} \).
    \item Proporcionar lecturas eficientes por rango de fechas y proyección de columnas.
    \item Gestionar versiones coherentes y metadatos consistentes.
\end{itemize}

\subsection{Contrato de interfaz y esquema de almacenamiento}

\paragraph{Entradas}
\begin{itemize}
    \item \texttt{features\_df(date,symbol,feature\_cols)}: \texttt{DataFrame} con:
        \begin{itemize}
            \item \texttt{date} \(: \mathbb{T}\),
            \item \texttt{symbol} \(: \mathcal{S}\),
            \item columnas de features numéricas \( \mathcal{F} \).
        \end{itemize}
    \item \texttt{feature\_dictionary}: definiciones de cada feature (nombre, tipo, descripción, lineage, etc.).
    \item \texttt{store\_config\_path}: configuración de particionado, codificación, compresión, rutas.
    \item \texttt{run\_id}: identificador único del snapshot.
\end{itemize}

\paragraph{Salidas}
\begin{itemize}
    \item \texttt{features/store/date=YYYY-MM-DD/part-*.parquet}:
        \begin{itemize}
            \item Particiones particionadas por \texttt{date} en formato Parquet.
            \item Clave lógica primaria: \texttt{(date, symbol)}.
        \end{itemize}
    \item \texttt{features/store/\_meta/feature\_dictionary\_<run\_id>.json}:
        \begin{itemize}
            \item Definiciones actualizadas de features para la versión \( \kappa \).
        \end{itemize}
    \item \texttt{features/store/\_meta/store\_manifest\_<run\_id>.json}:
        \begin{itemize}
            \item \texttt{run\_id}, \texttt{store\_config\_hash}, \texttt{features\_hash}, \texttt{start\_date}, \texttt{end\_date}.
        \end{itemize}
    \item \texttt{features/store/\_meta/schema\_<run\_id>.json}:
        \begin{itemize}
            \item Esquema de columnas (tipo, requerido/nullable).
        \end{itemize}
\end{itemize}

\subsection{Reglas de diseño y consistencia}

\paragraph{Clave lógica primaria}
\begin{itemize}
    \item \texttt{(date, symbol)} debe ser única en cada partición; se valida ausencia de duplicados.
    \item La unicidad asegura que cada señal \( X_{t}^{(f)}(s) \) esté bien definida y reproducible en training e inferencia.
\end{itemize}

\paragraph{Versionado y snapshots inmutables}
\begin{itemize}
    \item Cada \texttt{run\_id} produce un snapshot inmutable de \texttt{features/store/date=...}.
    \item Cambios de definición de feature se versionan en \texttt{feature\_dictionary} y \texttt{schema}.
\end{itemize}

\paragraph{Schema enforcement}
\begin{itemize}
    \item Al escribir, se valida:
        \begin{itemize}
            \item \texttt{required\_columns} presentes.
            \item Tipos numéricos coherentes (por ejemplo, \texttt{float64} para features).
        \end{itemize}
    \item Al leer, se pueden aplicar coerciones mínimas o rechazo de particiones corruptas.
\end{itemize}

\subsection{Operaciones core del feature store}

\subsubsection{Escritura de snapshots}
\begin{itemize}
    \item \texttt{write\_features\_snapshot(features\_df, feature\_dictionary, config, run\_id)}:
        \begin{enumerate}
            \item Validar PK y schema.
            \item Particionar por \texttt{date} y escribir Parquet.
            \item Escribir \texttt{feature\_dictionary\_<run\_id>.json}, \texttt{store\_manifest\_<run\_id>.json}, \texttt{schema\_<run\_id>.json}.
        \end{enumerate}
\end{itemize}

\subsubsection{Lectura de rangos de fechas}
\begin{itemize}
    \item \texttt{read\_features\_range(run\_id, start\_date, end\_date, columns=None)}:
        \begin{itemize}
            \item Filtra particiones \texttt{date} entre \texttt{start\_date} y \texttt{end\_date}.
            \item Aplica \texttt{column projection} para evitar lectura de features innecesarias.
        \end{itemize}
    \item Útil para training por ventanas o walk‑forward.
\end{itemize}

\subsubsection{Lectura de snapshot más reciente}
\begin{itemize}
    \item \texttt{read\_latest\_snapshot(columns=None)}:
        \begin{itemize}
            \item Resuelve el \texttt{run\_id} más reciente en el catálogo de versiones.
            \item Lee el snapshot completo o con proyección de columnas.
        \end{itemize}
    \item Útil para inference en producción o evaluación de último modelo.
\end{itemize}

\subsubsection{Validación de schema y metadatos}
\begin{itemize}
    \item \texttt{validate\_store\_schema(run\_id)}:
        \begin{itemize}
            \item Compara el esquema de Parquet con \texttt{schema\_<run\_id>.json}.
        \end{itemize}
    \item \texttt{list\_store\_versions(pattern=None)}:
        \begin{itemize}
            \item Lista run\_ids y metadatos de versiones disponibles.
        \end{itemize}
\end{itemize}

\subsection{Validaciones de calidad y consistencia}

\paragraph{Validaciones de escritura}
\begin{itemize}
    \item \textbf{PK uniqueness}: ausencia de duplicados \texttt{(date, symbol)}.
    \item \textbf{Schema coherence}: columnas requeridas presentes y tipos correctos.
    \item \textbf{Numeric sanity}: features numéricas sin corrupción de tipo.
    \item \textbf{Partition integrity}: particiones Parquet sin errores de formato.
\end{itemize}

\paragraph{Validaciones de lectura}
\begin{itemize}
    \item Comprobación de consistencia de esquema entre múltiples particiones.
    \item Manejo robusto de particiones faltantes o dañadas.
\end{itemize}

\subsection{Observabilidad y métricas operativas}

\paragraph{Métricas de escritura}
\begin{itemize}
    \item \texttt{n\_rows\_written}.
    \item \texttt{n\_dates\_written}.
    \item \texttt{n\_features}.
    \item \texttt{write\_read\_latency\_sec} para diagnóstico de performance de I/O.
\end{itemize}

\paragraph{Logging}
\begin{itemize}
    \item \texttt{INFO}: inicio de escritura, número de particiones, progreso por fecha.
    \item \texttt{WARNING}: fechas con baja densidad de features, warnings de schema débil.
    \item \texttt{ERROR}: fallo de escritura de partición o corrupción de manifest.
\end{itemize}

\subsection{CLI y uso en el pipeline global}

\paragraph{Interfaz de CLI recomendada}
\begin{verbatim}
python -m features.feature_store ^
  --input-path features/tmp/build_features_20260305_001.parquet ^
  --config-path configs/features/store.yaml ^
  --run-id feature_store_20260305_001
\end{verbatim}

\paragraph{Conexión con el sistema}
\begin{itemize}
    \item \texttt{models/*/train\_*.py}: lee snapshots por \texttt{run\_id} para entrenamiento reproducible.
    \item \texttt{backtest/run\_walkforward.py}: consume snapshots históricos versionados para simulación.
    \item \texttt{inference} ofrece acceso a \texttt{read\_latest\_snapshot} para producir señales en tiempo real.
\end{itemize}

\subsection{Definition of Done}

\begin{itemize}
    \item Feature store \textbf{estable, inmutable y consultable} por training e inferencia.
    \item Metadatos completos de versionado: \texttt{run\_id}, \texttt{feature\_dictionary}, \texttt{schema}, \texttt{manifest}.
    \item Garantía de unicidad \texttt{(date, symbol)} y consistencia de tipos numéricos.
\end{itemize}
