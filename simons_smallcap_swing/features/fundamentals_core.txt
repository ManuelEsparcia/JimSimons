\section{Módulo: \texttt{features/fundamentals\_core.py}}

\paragraph{Objetivo general}
Este módulo construye un conjunto de \textbf{features fundamentales base} (value, quality, profitability, leverage, turnover/growth) a partir de fundamentales PIT normalizados extraídos de EDGAR, produciendo ratios estandarizables por fecha y listos para ser integrados en un sistema de factoring de small caps con \textit{swing trading} multi‑factor. El objetivo es transformar datos contables complejos en señales robustas, comparables entre activos y coherentes con el marco de Fama–French, Carhart, y literatura de factores de valor calidad.

\subsection{Marco teórico y notación}

Sea \( \mathcal{S} \) el conjunto de símbolos y \( \mathcal{T} \) el rango de fechas. El módulo trabaja sobre:
\begin{itemize}
    \item \texttt{fundamentals\_pit}: vector de métricas contables \( \mathbf{F}^{(s,\texttt{asof})} \) con \texttt{asof\_date} \(\leq t\).
    \item \texttt{prices}: precio ajustado \( p_{t}^{(s)} \) y \texttt{share\_count} para convertir valores contables en ratios de valoración y mercado.
\end{itemize}

La \textbf{feature matrix} resultante es:
\[
X_{t}^{(f)}(s) = \phi^{(f)} \big( \texttt{fundamentals\_pit}(s, \tau \leq t), p_{t}^{(s)} \big),
\]
donde \( \phi^{(f)} \) es una función de ratio fundamental específica.

\subsection{Features fundamentales core}

\paragraph{Value / earnings yield}
\begin{itemize}
    \item \texttt{fund\_pe\_proxy} o \texttt{earnings\_yield}:
        \[
        \text{EY}_t^{(s)} = \frac{\text{Earnings}_{\text{ttm}}^{(s)}}{\text{Market Cap}_t^{(s)}}
        = \frac{\text{Net Income}_{\text{ttm}}^{(s)}}{p_{t}^{(s)} \cdot \text{SharesOut}_{t}^{(s)}},
        \]
        donde \texttt{ttm} indica “trailing twelve months” PIT, usando \texttt{asof\_date} válido.
\end{itemize}

\paragraph{Value / book–to–market}
\begin{itemize}
    \item \texttt{fund\_book\_to\_market}:
        \[
        \text{BtM}_t^{(s)} = \frac{\text{Book Value}_{\text{latest}}^{(s)}}{\text{Market Cap}_t^{(s)}}
        = \frac{\text{Total Equity}_{\text{latest}}^{(s)}}{p_{t}^{(s)} \cdot \text{SharesOut}_{t}^{(s)}},
        \]
        con \texttt{Total Equity} ajustado por data PIT, dejando fuera quiebras o anomalias de capital contable.
\end{itemize}

\paragraph{Profitability / quality}
\begin{itemize}
    \item \texttt{fund\_profit\_margin}:
        \[
        \text{PM}_t^{(s)} = \frac{\text{Net Income}_{\text{ttm}}^{(s)}}{\text{Revenue}_{\text{ttm}}^{(s)}},
        \]
        captura calidad de ganancias y disciplina de costes.
    \item \texttt{fund\_roa} (Return on Assets):
        \[
        \text{ROA}_t^{(s)} = \frac{\text{Net Income}_{\text{ttm}}^{(s)}}{\text{Total Assets}_{\text{latest}}^{(s)}}.
        \]
    \item \texttt{fund\_roe\_proxy}:
        \[
        \text{ROE}_t^{(s)} \approx \frac{\text{Net Income}_{\text{ttm}}^{(s)}}{\text{Average Equity}_{\text{window}}},
        \]
        estimado con media de equity en ventana PIT-compatible.
\end{itemize}

\paragraph{Leverage}
\begin{itemize}
    \item \texttt{fund\_leverage\_ratio}:
        \[
        \text{Leverage}_t^{(s)} = \frac{\text{Total Debt}_{\text{latest}}^{(s)}}{\text{Total Assets}_{\text{latest}}^{(s)}},
        \]
        midiendo exposición de balance a riesgo de crédito y volatilidad de estructura de capital.
\end{itemize}

\paragraph{Asset turnover / growth}
\begin{itemize}
    \item \texttt{fund\_asset\_turnover}:
        \[
        \text{Turnover}_t^{(s)} = \frac{\text{Revenue}_{\text{ttm}}^{(s)}}{\text{Total Assets}_{\text{latest}}^{(s)}},
        \]
        asociado a eficiencia de capital y potencial de crecimiento orgánico.
\end{itemize}

\subsection{Reglas PIT y control de staleness}

\paragraph{Disponibilidad PIT}
\begin{itemize}
    \item Solo se usan métricas con \texttt{asof\_date} \(\leq t\), respetando el calendario de reporting de small caps y el \texttt{decision\_lag} configurado.
    \item Para cada fecha \( t \), se asigna a cada símbolo \( s \) el \texttt{asof} más reciente disponible.
\end{itemize}

\paragraph{Forward–fill limitado}
\begin{itemize}
    \item Si una métrica fundamental no está disponible en \( t \), se usa forward–fill por máximo \texttt{max\_staleness} (por ejemplo, 90–180 días).
    \item Tras exceder \texttt{max\_staleness}, se considera \texttt{stale\_severely} y se fuerza a \texttt{null} o se excluye según política.
\end{itemize}

\paragraph{Política de tratamiento}
\begin{itemize}
    \item \texttt{normative\_policy}: advertir pero no imputar si se viola \texttt{max\_staleness} de métricas críticas.
    \item \texttt{fallback\_policy}: para denominadores críticos, usar valores anteriores no nulos o imputación cross‑sectional si permitido.
\end{itemize}

\subsection{Validaciones y rangos de sanidad}

\paragraph{Reglas de validación}
\begin{itemize}
    \item \textbf{Denominadores no nulos}:
        \[
        \text{Denominator}^{(s)} \neq 0 \quad \text{o} \quad \text{fallback\_value} \text{ configurado}.
        \]
    \item \textbf{Rangos de ratios plausibles}:
        \begin{itemize}
            \item \texttt{Profit margin} > \(-100\%\) y < \(100\%\) típicamente, salvo casos excepcionales (excluidos).
            \item \texttt{ROA} y \texttt{ROE} limitados por zonas de outliers.
        \end{itemize}
    \item \textbf{Cobertura mínima por métrica}:
        \[
        \rho_{\text{cov}}^{(f)} = \frac{1}{|\mathcal{T}|} \sum_{t \in \mathcal{T}} \rho_{\text{cov}}^{(f,t)} \geq \texttt{min\_coverage\_pct},
        \]
        donde cada \(\rho_{\text{cov}}^{(f,t)}\) es el porcentaje de símbolos con observación.
\end{itemize}

\subsection{Flujo de procesamiento}

\paragraph{Alto nivel}
1. \texttt{load\_fundamentals\_pit(fundamentals\_pit\_path, prices\_path, cfg)}:
    - Carga datos de EDGAR PIT y precios ajustados.
2. \texttt{compute\_core\_ratios(...)}:
    - Calcula \texttt{fund\_pe\_proxy}, \texttt{fund\_book\_to\_market}, \texttt{fund\_profit\_margin}, \texttt{fund\_roa}, \texttt{fund\_roe\_proxy}, \texttt{fund\_leverage\_ratio}, \texttt{fund\_asset\_turnover}.
3. \texttt{apply\_staleness\_controls(...)}:
    - Aplica reglas de \texttt{max\_staleness} y exclusiones severas.
4. \texttt{apply\_outlier\_filters(...)}:
    - Winsorización o corte de ratios extremos.
5. \texttt{persist\_outputs(...)}:
    - Guarda la matriz de features, \texttt{summary\_run\_id.json}, \texttt{manifest\_run\_id.json}.

\subsection{Salidas y estructura de almacenamiento}

\paragraph{Outputs principales}
\begin{itemize}
    \item \texttt{features/fundamentals\_core/date={yyyy-mm-dd}/part-*.parquet}:
        \begin{itemize}
            \item Columnas base: \texttt{date}, \texttt{symbol}, \texttt{run\_id}.
            \item Features de fundamentales:
                \begin{itemize}
                    \item \texttt{fund\_pe\_proxy}, \texttt{fund\_book\_to\_market},
                    \texttt{fund\_profit\_margin\_core}, \texttt{fund\_roa}, \texttt{fund\_roe\_proxy},
                    \texttt{fund\_leverage\_ratio}, \texttt{fund\_asset\_turnover}.
                \end{itemize}
        \end{itemize}
    \item \texttt{features/fundamentals\_core/summary\_\{run\_id\}.json}:
        \begin{itemize}
            \item \texttt{coverage\_rate}, \texttt{mean}, \texttt{p50}, \texttt{p99}, \texttt{n\_stale\_exclusions}.
        \end{itemize}
    \item \texttt{features/fundamentals\_core/manifest\_\{run\_id\}.json}:
        \begin{itemize}
            \item \texttt{run\_id}, \texttt{config\_hash}, \texttt{fundamentals\_pit\_hash}, \texttt{prices\_path}.
        \end{itemize}
\end{itemize}

\subsection{CLI y uso en el pipeline global}

\paragraph{CLI recomendada}
\begin{verbatim}
python -m features.fundamentals_core ^
  --fundamentals-pit-path data/edgar/pit ^
  --prices-path data/price/adjusted/mode=dual_output ^
  --config-path configs/features/fundamentals_core.yaml ^
  --run-id fundamentals_core_20260305_001
\end{verbatim}

\paragraph{Uso downstream}
\begin{itemize}
    \item \texttt{features/build\_features.py}: integra ratios \texttt{fund\_*} como componentes de familias de features fundamentales.
    \item \texttt{models/*/train\_*.py}: construye factores de value–quality–leverage.
    \item \texttt{backtest/run\_walkforward.py}: exploita rankings de value/quality para construir long–short portfolios de small caps.
\end{itemize}

\subsection{Definition of Done}

\begin{itemize}
    \item Conjunto de \textbf{features fundamentales PIT} listo para estandarización cross‑sectional y modelado.
    \item Ratios bien definidos matemáticamente, con tratamiento de staleness controlado.
    \item Salidas almacenadas de forma versionada, con metadata completa y cobertura adecuada para seguir el pipeline de \texttt{feature\_qc} y \texttt{cross\_sectional}.
\end{itemize}
