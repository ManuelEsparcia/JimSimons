\section{Módulo: \texttt{features/feature\_qc.py}}

\paragraph{Objetivo general}
Este módulo implementa un \textbf{sistema de calidad de datos} para el feature store, validando integridad estructural, cobertura, estabilidad temporal, ausencia de leakage y control de colinealidad extrema, antes de utilizar las features en modelado o backtest. El objetivo es emitir un \textbf{veredicto gateado} (pass/warn/fail) y recomendaciones accionables sobre qué features mantener, revisar o descartar, asegurando que el sistema de swing trading multi‑factor opere sobre un dataset metodológicamente limpio.

\subsection{Marco teórico y notación}

Sea \( \mathcal{F} \) el conjunto de features en el store y \( \mathcal{T} \subseteq \mathbb{T} \) el rango de fechas de análisis. Definimos:
\begin{itemize}
    \item \( X_{t}^{(f)} \in \mathbb{R}^{|\mathcal{S}_t|} \): vector de feature \( f \in \mathcal{F} \) en fecha \( t \in \mathcal{T} \).
    \item \( \mathbf{X}^{(f)} \in \mathbb{R}^{|\mathcal{T}| \times |\mathcal{S}_t|} \): serie temporal de la feature \( f \) cruzando fechas y símbolos.
    \item \( y_{t}^{(s)} \in \mathbb{R} \): label objetivo (retorno de swing o factor de riesgo), si disponible.
\end{itemize}

El objetivo de \texttt{feature\_qc.py} es calcular métricas de calidad \( \mathcal{Q}^{(f)} \) para cada feature, y combinaciones de ellas para el conjunto global, y decidir:
\[
\text{Gate}(\mathcal{Q}) \in \{ \text{pass}, \text{warn}, \text{fail} \}.
\]

\subsection{Checks core y definición de métricas}

\subsubsection{Integridad estructural}

\paragraph{Validación de schema}
\begin{itemize}
    \item \texttt{schema\_expected} vs \texttt{schema\_observed}: chequeo de nombres de columnas, tipos (features numéricas, flags categóricos, fecha/simbolo).
    \item Fallos en schema se clasifican como \textbf{críticos} (gate 1).
\end{itemize}

\paragraph{Duplicados y consistencia}
\begin{itemize}
    \item Verificar ausencia de duplicados en \texttt{(date, symbol)}.
    \item Esta condición es necesaria para que los estadísticos cross‑sectionales sean bien definidos.
\end{itemize}

\subsubsection{Cobertura}

\paragraph{Definiciones}
\begin{enumerate}
    \item \texttt{null\_rate} global por feature:
        \[
        \rho_{\text{null}}^{(f)} = \frac{1}{|\mathcal{T}|} \sum_{t \in \mathcal{T}}
        \frac{ \sum_{s \in \mathcal{S}_t} \mathbb{I} \big( X_{t}^{(f)}(s) = \text{null} \big) }{|\mathcal{S}_t|}.
        \]
    \item \texttt{coverage\_rate} global:
        \[
        \rho_{\text{cov}}^{(f)} = 1 - \rho_{\text{null}}^{(f)}.
        \]
    \item \texttt{coverage\_rate} por fecha:
        \[
        \rho_{\text{cov}}^{(f,t)} = \frac{ \sum_{s \in \mathcal{S}_t} \mathbb{I} \big( X_{t}^{(f)}(s) \neq \text{null} \big) }{|\mathcal{S}_t|}.
        \]
\end{enumerate}

\paragraph{Umbrales}
\begin{itemize}
    \item Gate 2: \texttt{pct\_features\_with\_coverage\_above\_threshold} debe superar un umbral configurable.
\end{itemize}

\subsubsection{Sanidad numérica}

\paragraph{Detección de valores anómalos}
\begin{itemize}
    \item \texttt{inf}, \texttt{-inf}, \texttt{NaN} no permitidos salvo en política explícita.
    \item \texttt{constant\_feature}:
        \[
        \text{Var} \big( \mathbf{X}^{(f)} \big) < \epsilon
        \]
        para algún umbral \( \epsilon > 0 \), se considera feature de baja utilidad predictiva.
    \item \texttt{outlier\_rate}:
        \[
        \rho_{\text{out}}^{(f,t)} = \frac{1}{|\mathcal{S}_t|} \sum_{s} \mathbb{I} \big( |X_{t}^{(f)}(s) - \mu_{t}^{(f)}| > \gamma \sigma_{t}^{(f)} \big),
        \]
        donde \( \gamma \) es configurable (por ejemplo, 3–5 desviaciones estándar).
\end{itemize}

\subsubsection{Dependencias entre features}

\paragraph{Correlación y multicolinealidad}
\begin{itemize}
    \item Correlación par a par:
        \[
        \rho_{ij,t} = \text{corr} \big( X_{t}^{(f_i)}, X_{t}^{(f_j)} \big).
        \]
        Si \( |\rho_{ij,t}| > \rho_{\text{high}} \) (por ejemplo, 0.98) durante un porcentaje significativo de fechas, se levanta flag de riesgo de colinealidad.
    \item VIF (Variance Inflation Factor) aproximado para bloques de features:
        \[
        \text{VIF}^{(j)} = \frac{1}{1 - R^2_j},
        \]
        donde \( R^2_j \) es el coeficiente de determinación de regresión de la feature \( j \) sobre el resto de un grupo de bloques correlados. \cite{marquardt1970}
\end{itemize}

\paragraph{Gate 5 – Colinealidad}
\begin{itemize}
    \item \texttt{pct\_features\_high\_corr} > umbral configurable \(\Rightarrow\) \texttt{warn} o \texttt{fail}.
\end{itemize}

\subsection{Leakage temporal y pruebas de fuga de datos}

\paragraph{Shift test (leakage proxy)}
Cuando labels están disponibles:
\begin{itemize}
    \item \texttt{leakage\_score\_forward}:
        \[
        \rho_{t}^{(f,\text{fwd})} = \text{corr} \big( X_{t}^{(f)}, y_{t+1}^{(\cdot)} \big).
        \]
    \item \texttt{leakage\_score\_lagged}:
        \[
        \rho_{t}^{(f,\text{lag})} = \text{corr} \big( X_{t}^{(f)}, y_{t-\tau}^{(\cdot)} \big),
        \]
        con \( \tau \in \mathbb{N}^+ \) correspondiente al horizonte de decision\_lag.
\end{itemize}
Si \( \rho_{t}^{(f,\text{fwd})} \gg \rho_{t}^{(f,\text{lag})} \) de forma consistente, se considera señal de possible leakage.

\paragraph{Availability test}
\begin{itemize}
    \item Verificar que el timestamp de origen de la feature (en metadata) no sea posterior a \texttt{decision\_cut} ajustado según \texttt{decision\_lag\_days}.
    \item Register \texttt{leakage\_score\_time} basado en discrepancias de timestamp.
\end{itemize}

\paragraph{Availability de labels}
\begin{itemize}
    \item El uso de labels en QC sigue las mismas reglas PIT que el pipeline de entrenamiento y backtest; no se mezcla horizonte futuro en el análisis de QC.
\end{itemize}

\subsection{Análisis de drift de features}

\paragraph{Problemática}
El drift temporal de distribuciones de features afecta la estabilidad de modelos de swing trading, especialmente en small caps con regímenes de mercado cambiando (Harvey et al., 2017; \cite{harvey2017}).

\paragraph{Métricas de drift}
\begin{itemize}
    \item PSI (Population Stability Index):
        \[
        \text{PSI}^{(f)} = \sum_{j} \big( p_{j,\text{curr}}^{(f)} - p_{j,\text{ref}}^{(f)} \big)
        \log \bigg( \frac{p_{j,\text{curr}}^{(f)}}{p_{j,\text{ref}}^{(f)}} \bigg),
        \]
        donde \( p_{j,\text{ref}}^{(f)} \) y \( p_{j,\text{curr}}^{(f)} \) son las proporciones de bins en referencia y ventana actual respectivamente.
    \item KS (Kolmogorov–Smirnov):
        \[
        \text{KS}^{(f)} = \sup_{x} \big| F_{\text{curr}}^{(f)}(x) - F_{\text{ref}}^{(f)}(x) \big|,
        \]
        usando CDF empíricas de \textbf{feature\_f} en referencia y ventana actual.
    \item Jensen–Shannon (JS):
        \[
        \text{JS}^{(f)} = \frac{1}{2} \text{KL}(P_{\text{curr}}^{(f)} \| M^{(f)}) +
        \frac{1}{2} \text{KL}(P_{\text{ref}}^{(f)} \| M^{(f)}),
        \]
        donde \( M^{(f)} = \frac{1}{2}(P_{\text{curr}}^{(f)} + P_{\text{ref}}^{(f)}) \).
\end{itemize}

\paragraph{Gate 4 – Drift}
\begin{itemize}
    \item \texttt{drift\_score\_aggregated} (por ejemplo, promedio de PSI/KS por feature) ≤ umbral \(\Rightarrow\) \texttt{pass}.
\end{itemize}

\subsection{Reglas PIT y diseño de gates}

\paragraph{Reglas PIT}
\begin{itemize}
    \item Todos los checks deben respetar el \texttt{decision\_lag} configurado en la pipeline.
    \item Comparaciones de drift y leakage se hacen por fecha o por ventana de referencia, sin mezclar todo el panel sin control temporal.
    \item Se guarda el \texttt{run\_id}, hash de \texttt{config} y \texttt{feature\_dictionary} para trazabilidad.
\end{itemize}

\paragraph{Gates de aceptación}
\begin{itemize}
    \item \textbf{Gate 1 – Integridad}: schema válido y 0 duplicados en \texttt{(date, symbol)} → \textbf{crítico}.
    \item \textbf{Gate 2 – Cobertura}: \texttt{pct\_features\_with\_coverage\_above\_threshold} suficiente.
    \item \textbf{Gate 3 – Leakage}: \texttt{leakage\_score\_aggregated} ≤ umbral.
    \item \textbf{Gate 4 – Drift}: \texttt{drift\_score\_aggregated} ≤ umbral.
    \item \textbf{Gate 5 – Colinealidad}: \texttt{pct\_features\_high\_corr} ≤ umbral.
\end{itemize}
\begin{itemize}
    \item \texttt{pass}: todos los gates críticos satisfechos.
    \item \texttt{warn}: fallos menores dentro de tolerancia.
    \item \texttt{fail}: cualquier gate crítico incumplido.
\end{itemize}

\subsection{Flujo de proceso y estructura de funciones}

\paragraph{Alto nivel}
1. Cargar config, store, y diccionario de features.
2. Validar estructura base (schema, claves, run\_id, fechas).
3. Computar cobertura:
    \begin{itemize}
        \item \texttt{compute\_coverage\_metrics(df\_feat, cfg)}.
    \end{itemize}
4. Computar sanidad numérica:
    \begin{itemize}
        \item \texttt{compute\_numeric\_sanity\_metrics(df\_feat, cfg)}.
    \end{itemize}
5. Computar dependencias:
    \begin{itemize}
        \item \texttt{compute\_collinearity\_metrics(df\_feat, cfg)}.
    \end{itemize}
6. Computar pruebas de leakage:
    \begin{itemize}
        \item \texttt{compute\_leakage\_proxies(df\_feat, df\_labels, cfg)}.
    \end{itemize}
7. Computar drift:
    \begin{itemize}
        \item \texttt{compute\_drift\_metrics(df\_feat, df\_ref, cfg)}.
    \end{itemize}
8. Agregar puntuaciones por feature:
    \begin{itemize}
        \item \texttt{aggregate\_qc\_scores(df\_metrics, cfg)}.
    \end{itemize}
9. Aplicar gates:
    \begin{itemize}
        \item \texttt{apply\_feature\_gates(df\_metrics, cfg)}.
    \end{itemize}
10. Persistir:
    \begin{itemize}
        \item \texttt{persist\_qc\_reports(...)}.
    \end{itemize}
\end{itemize}

\subsection{Manejo de casos especiales}

\begin{itemize}
    \item \texttt{Feature nueva sin baseline}:
        \begin{itemize}
            \item Marcar como \texttt{new\_feature} y excluir de gate de drift o usar una ventana de referencia corta.
        \end{itemize}
    \item \texttt{Feature sparse\_expected}:
        \begin{itemize}
            \item Permitir alta tasa de nulos si está etiquetada como tal en el \texttt{feature\_dictionary}.
        \end{itemize}
    \item \texttt{Cambio de definición de feature}:
        \begin{itemize}
            \item Detectar por cambio de \texttt{formula\_hash} o \texttt{lineage\_hash} y evitar comparaciones de drift directas entre versiones.
        \end{itemize}
    \item \texttt{Universo pequeño}:
        \begin{itemize}
            \item Relajar tests estadísticos y elevar \texttt{warning} de baja potencia estadística.
        \end{itemize}
\end{itemize}

\subsection{Reportes de salida y observabilidad}

\paragraph{Artefactos principales}
\begin{itemize}
    \item \texttt{features/qc/qc\_summary.json}:
        \begin{itemize}
            \item \texttt{qc\_global\_score}, \texttt{gate\_status}, \texttt{n\_critical\_failures}.
        \end{itemize}
    \item \texttt{features/qc/qc\_feature\_level.parquet}:
        \begin{itemize}
            \item \texttt{feature\_name}, \texttt{null\_rate\_global}, \texttt{coverage\_rate\_global}, \texttt{p01}, \texttt{p50}, \texttt{p99}, \texttt{outlier\_rate}, \texttt{drift\_score}, \texttt{leakage\_score}, \texttt{collinearity\_risk}, \texttt{gate\_status}, \texttt{recommended\_action} (keep/drop/review).
        \end{itemize}
    \item \texttt{features/qc/qc\_date\_level.parquet}:
        \begin{itemize}
            \item Métricas agregadas por fecha (coverage, drift, etc.).
        \end{itemize}
    \item \texttt{features/qc/qc\_failures.parquet}:
        \begin{itemize}
            \item Registros de features/fechas que fallan checks.
        \end{itemize}
    \item \texttt{features/qc/qc\_drift\_report.parquet}.
    \item \texttt{features/qc/qc\_manifest.json} y \texttt{features/qc/qc\_flags.json}.
\end{itemize}

\paragraph{Logging y métricas}
\begin{itemize}
    \item \texttt{INFO}: progreso por bloque de checks.
    \item \texttt{WARNING}: features/fechas con riesgo moderado.
    \item \texttt{ERROR}: fallos críticos (schema, leakage severo).
    \item Métricas operativas: \texttt{n\_features\_checked}, \texttt{n\_features\_failed}, \texttt{n\_features\_warn}, \
