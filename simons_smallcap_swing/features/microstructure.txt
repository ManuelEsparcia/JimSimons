\section{Módulo: \texttt{features/microstructure.py}}

\paragraph{Objetivo general}
Este módulo construye \textbf{features de microestructura} a nivel de small cap, generando proxies de \textit{spread implícito}, \textit{gap overnight}, \textit{rango intradía} y \textit{amihud‑like liquidity} a partir de datos OHLCV diarios. El objetivo es cuantificar la \textit{fricción de trading} y la \textit{dinámica de precio de corto plazo}, para modular señales de momentum y reversión de swing trading y ajustar la exposición de risk de tamaño/microestructura.

\subsection{Marco teórico y notación}

Sea \( \mathcal{S} \) el conjunto de símbolos y \( \mathcal{T} \) el rango de fechas de trading. Para cada símbolo \( s \in \mathcal{S} \) y fecha \( t \in \mathcal{T} \), se tienen:
\begin{itemize}
    \item \texttt{open\_adj}_t^{(s)}, \texttt{high\_adj}_t^{(s)}, \texttt{low\_adj}_t^{(s)}, \texttt{close\_adj}_t^{(s)}.
    \item \texttt{volume\_adj}_t^{(s)}.
    \item \( r_{t}^{(s)} = \log p_{t}^{(s)} - \log p_{t-1}^{(s)} \), retorno logarítmico diario.
\end{itemize}

Las features de microestructura capturan:
\begin{itemize}
    \item \textit{Coste de liquidez} (spread proxy, \texttt{amihud}).
    \item \textit{Impacto de gap} y \textit{volatilidad intradía}.
    \item \textit{Shocks de volumen} y \textit{reversión de corto plazo}.
\end{itemize}

\subsection{Features de microestructura sugeridas}

\subsubsection{Microspread proxy}
\begin{itemize}
    \item \texttt{micro\_spread\_proxy}:
        \[
        \text{Spread}_t^{(s)} = \frac{\text{high\_adj}_t^{(s)} - \text{low\_adj}_t^{(s)}}{\text{close\_adj}_t^{(s)}},
        \]
        proxy relativo del rango intradía, relacionado con el spread implícito (Roll, 1984). \cite{roll1984}
\end{itemize}

\subsubsection{Gap de apertura}
\begin{itemize}
    \item \texttt{micro\_gap\_open}:
        \[
        \text{Gap}_t^{(s)} = \frac{\text{open\_adj}_t^{(s)} - \text{close\_adj}_{t-1}^{(s)}}{\text{close\_adj}_{t-1}^{(s)}},
        \]
        mide \textit{overnight news impact} y posible \textit{gap risk} en small caps.
\end{itemize}

\subsubsection{Rango de sesión (\texttt{HL\_range})}
\begin{itemize}
    \item \texttt{micro\_hl\_range}:
        \[
        \text{Range}_t^{(s)} = \log \Big( \frac{\text{high\_adj}_t^{(s)}}{\text{low\_adj}_t^{(s)}} \Big),
        \]
        mide la volatilidad intradía como medida de \textit{price discovery} diario.
\end{itemize}

\subsubsection{Proxy de Amihud}
\begin{itemize}
    \item \texttt{micro\_amihud\_proxy}:
        \[
        \text{Amihud}_t^{(s)} = \frac{|r_{t}^{(s)}|}{\text{volume\_adj}_t^{(s)} \cdot p_{t}^{(s)} + \epsilon},
        \]
        donde:
        \begin{itemize}
            \item numerador: magnitud de retorno diario.
            \item denominador: volumen en valor nominal (USD).
        \end{itemize}
        Este proxy capta \textit{precio por unidad de volumen} o \textit{illiquidity} (Amihud, 2002). \cite{amihud2002}
\end{itemize}

\subsubsection{Shock de volumen}
\begin{itemize}
    \item \texttt{micro\_volume\_shock}:
        \[
        \text{VolShock}_t^{(s)} = \frac{v_{t}^{(s)} - \mu_{t,\text{window}}^{(v,s)}}{\sigma_{t,\text{window}}^{(v,s)}},
        \]
        donde \(\mu_{t,\text{window}}^{(v,s)}\) y \(\sigma_{t,\text{window}}^{(v,s)}\) son media y volatilidad del volumen en una ventana histórica PIT segura.
\end{itemize}

\subsubsection{Reversión de retorno de 1 día}
\begin{itemize}
    \item \texttt{micro\_return\_reversal\_1d}:
        \[
        \text{Reversal}_t^{(s)} = - r_{t}^{(s)},
        \]
        captura reversión de corto plazo asociada a fricción de mercado y market impact.
\end{itemize}

\subsection{Reglas PIT y fiabilidad temporal}

\paragraph{Uso de información histórica}
\begin{itemize}
    \item Todas las métricas de microestructura se construyen usando \textbf{la sesión diaria actual} y el \textbf{historial de fechas anteriores}.
    \item No se usa ningún precio futuro (cierre de \( t+1 \)) o \texttt{future\_window} en las fórmulas.
\end{itemize}

\paragraph{Ejemplo de ventana de volumen}
\begin{itemize}
    \item \texttt{micro\_volume\_shock} se calcula sobre \texttt{volumen ajustado} solo \(\leq t-1\), con \texttt{window} controlada por configuración.
\end{itemize}

\subsection{Flujo de procesamiento}

\paragraph{Alto nivel}
1. \texttt{load\_prices\_adjusted(prices\_path, config)}:
    - Carga \texttt{OHLCV} ajustado por \texttt{date} y \texttt{symbol}.
2. \texttt{compute\_micro\_features(...)}:
    - Calcula \texttt{micro\_spread\_proxy}, \texttt{micro\_gap\_open}, \texttt{micro\_hl\_range}, \texttt{micro\_amihud\_proxy}, \texttt{micro\_volume\_shock}, \texttt{micro\_return\_reversal\_1d} por símbolo/fecha.
3. \texttt{apply\_clipping\_winsor(...)}:
    - Aplica clipping o winsorización a percentiles \([p_{\min}, p_{\max}]\) configurables para outliers.
4. \texttt{persist\_micro\_features(...)}:
    - Guarda \texttt{features/microstructure/date=...} junto con \texttt{summary\_run\_id.json} y \texttt{manifest\_run\_id.json}.

\subsection{Validaciones de sanidad y consistencia}

\paragraph{Valores finitos}
\begin{itemize}
    \item \texttt{No inf, -inf} en \texttt{micro\_amihud\_proxy} o \texttt{micro\_volume\_shock}.
    \item Comprobación de división por cero mediante \(\epsilon\) o denominadores razonables.
\end{itemize}

\paragraph{Rangos plausibles}
\begin{itemize}
    \item \texttt{micro\_spread\_proxy} típico en \([0.001, 0.2]\) para small caps líquidos.
    \item \texttt{micro\_amihud\_proxy} limitado por cotas empíricas de \textit{illiquidity}.
\end{itemize}

\paragraph{Cobertura mínima}
\begin{itemize}
    \item \texttt{coverage\_rate} por feature:
        \[
        \rho_{\text{cov}}^{(f_{\text{micro}})} \geq \texttt{min\_coverage\_pct}.
        \]
\end{itemize}

\subsection{Salidas y estructura de almacenamiento}

\paragraph{Outputs principales}
\begin{itemize}
    \item \texttt{features/microstructure/date=\{yyyy-mm-dd\}/part-*.parquet}:
        \begin{itemize}
            \item \texttt{date}, \texttt{symbol}, \texttt{run\_id}.
            \item Columnas de microestructura:
                \begin{itemize}
                    \item \texttt{micro\_spread\_proxy}, \texttt{micro\_gap\_open},
                    \texttt{micro\_hl\_range}, \texttt{micro\_amihud\_proxy},
                    \texttt{micro\_volume\_shock}, \texttt{micro\_return\_reversal\_1d}.
                \end{itemize}
        \end{itemize}
    \item \texttt{features/microstructure/summary\_\{run\_id\}.json}:
        \begin{itemize}
            \item \texttt{mean}, \texttt{p50}, \texttt{p99}, \texttt{coverage\_rate} por feature.
        \end{itemize}
    \item \texttt{features/microstructure/manifest\_\{run\_id\}.json}:
        \begin{itemize}
            \item \texttt{run\_id}, \texttt{microstructure\_config\_hash}, \texttt{prices\_path\_hash}.
        \end{itemize}
\end{itemize}

\subsection{CLI y uso en el pipeline global}

\paragraph{CLI recomendada}
\begin{verbatim}
python -m features.microstructure ^
  --prices-path data/price/adjusted/mode=dual_output ^
  --config-path configs/features/microstructure.yaml ^
  --run-id microstructure_20260305_001
\end{verbatim}

\paragraph{Uso downstream}
\begin{itemize}
    \item \texttt{features/build\_features.py}: integra \texttt{micro\_spread\_proxy}, \texttt{micro\_amihud\_proxy} y \texttt{micro\_gap\_open} como factores de \textit{liquidez} y \textit{risk de microestructura}.
    \item \texttt{feature\_qc.py}: valida cobertura y drift de features de microestructura.
    \item \texttt{backtest/run\_walkforward.py}: explora \textit{take‑profit} y \textit{stop} adaptados a volatilidad intradía y \textit{gap risk}.
\end{itemize}

\subsection{Definition of Done}

\begin{itemize}
    \item Conjunto de \textbf{micro‑features} PIT listos para estandarización y análisis de calidad.
    \item Diseño sin leakage temporal y con chequeo de sanidad numérica.
    \item Integración limpia con \texttt{feature\_qc} y \texttt{build\_features} como familia de señales líquidas y de fricción.
\end{itemize}
